   {

               slint::cbindgen_private::Slice<uint8_t> { const_cast<uint8_t *>(data.data()),

               slint::cbindgen_private::Slice<uint8_t> {
                       const_cast<uint8_t *>(reinterpret_cast<const uint8_t *>(extension.data())),
                       extension.size() },
               &img);
       return Image(img);
   }
