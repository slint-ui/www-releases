17/// Extension trait, adding JSON serialization methods
18pub trait JsonExt
19where
20    Self: Sized,
21{
22    /// Convert to a JSON object
23    fn to_json(&self) -> Result<serde_json::Value, String>;
24    /// Convert to a JSON-encoded string
25    fn to_json_string(&self) -> Result<String, String>;
26    /// Convert to JSON object to `Self`
27    fn from_json(t: &langtype::Type, value: &serde_json::Value) -> Result<Self, String>;
28    /// Convert to JSON encoded string to `Self`
29    fn from_json_str(t: &langtype::Type, value: &str) -> Result<Self, String>;
30}
31
32impl JsonExt for crate::Value {
33    fn to_json(&self) -> Result<serde_json::Value, String> {
34        value_to_json(self)
35    }
36
37    fn to_json_string(&self) -> Result<String, String> {
38        value_to_json_string(self)
39    }
40
41    fn from_json(t: &langtype::Type, value: &serde_json::Value) -> Result<Self, String> {
42        value_from_json(t, value)
43    }
44
45    fn from_json_str(t: &langtype::Type, value: &str) -> Result<Self, String> {
46        value_from_json_str(t, value)
47    }
48}

50/// Create a `Value` from a JSON Value
51pub fn value_from_json(t: &langtype::Type, v: &serde_json::Value) -> Result<Value, String> {
52    use smol_str::ToSmolStr;
53
54    fn string_to_color(s: &str) -> Option<i_slint_core::Color> {
55        i_slint_compiler::literals::parse_color_literal(s).map(Color::from_argb_encoded)
56    }
57
58    match v {
59        serde_json::Value::Null => Ok(Value::Void),
60        serde_json::Value::Bool(b) => Ok((*b).into()),
61        serde_json::Value::Number(n) => Ok(Value::Number(n.as_f64().unwrap_or(f64::NAN))),
62        serde_json::Value::String(s) => match t {
63            langtype::Type::Enumeration(e) => {
64                let s = s.to_smolstr();
65                if e.values.contains(&s) {
66                    Ok(Value::EnumerationValue(e.name.to_string(), s.into()))
67                } else {
68                    Err(format!("Unexpected value for enum '{}': {}", e.name, s))
69                }
70            }
71            langtype::Type::Color => {
72                if let Some(c) = string_to_color(s) {
73                    Ok(Value::Brush(i_slint_core::Brush::SolidColor(c)))
74                } else {
75                    Err(format!("Failed to parse color: {s}"))
76                }
77            }
78            langtype::Type::String => Ok(SharedString::from(s.as_str()).into()),
79            langtype::Type::Image => match Image::load_from_path(std::path::Path::new(s)) {
80                Ok(image) => Ok(image.into()),
81                Err(e) => Err(format!("Failed to load image from path: {s}: {e}")),
82            },
83            langtype::Type::Brush => {
84                if s.starts_with('#') {
85                    if let Some(c) = string_to_color(s) {
86                        Ok(Value::Brush(i_slint_core::Brush::SolidColor(c)))
87                    } else {
88                        Err(format!("Failed to parse color value {s}"))
89                    }
90                } else {
91                    Err("Brush kind not supported".into())
92                }
93            }
94            _ => Err("Value type not supported".into()),
95        },
96        serde_json::Value::Array(array) => match t {
97            langtype::Type::Array(it) => {
98                Ok(Value::Model(ModelRc::new(i_slint_core::model::SharedVectorModel::from(
99                    array
100                        .iter()
101                        .map(|v| value_from_json(it, v))
102                        .collect::<Result<SharedVector<Value>, String>>()?,
103                ))))
104            }
105            _ => Err("Got an array where none was expected".into()),
106        },
107        serde_json::Value::Object(obj) => match t {
108            langtype::Type::Struct(s) => Ok(crate::Struct(
109                obj.iter()
110                    .map(|(k, v)| {
111                        let k = k.to_smolstr();
112                        match s.fields.get(&k) {
113                            Some(t) => value_from_json(t, v).map(|v| (k.to_string(), v)),
114                            None => Err(format!("Found unknown field in struct: {k}")),
115                        }
116                    })
117                    .collect::<Result<HashMap<String, Value>, _>>()?,
118            )
119            .into()),
120            _ => Err("Got a struct where none was expected".into()),
121        },
122    }
123}

125/// Create a `Value` from a JSON string
126pub fn value_from_json_str(t: &langtype::Type, v: &str) -> Result<Value, String> {
127    let value = serde_json::from_str(v).map_err(|e| format!("Failed to parse JSON: {e}"))?;
128    Value::from_json(t, &value)
129}

131/// Write the `Value` out into a JSON value
132pub fn value_to_json(value: &Value) -> Result<serde_json::Value, String> {
133    fn color_to_string(color: &Color) -> String {
134        let a = color.alpha();
135        let r = color.red();
136        let g = color.green();
137        let b = color.blue();
138
139        format!("#{r:02x}{g:02x}{b:02x}{a:02x}")
140    }
141
142    match value {
143        Value::Void => Ok(serde_json::Value::Null),
144        Value::Bool(b) => Ok((*b).into()),
145        Value::Number(n) => {
146            let r = if *n == n.round() {
147                if *n >= 0.0 {
148                    serde_json::Number::from_u128(*n as u128)
149                } else {
150                    serde_json::Number::from_i128(*n as i128)
151                }
152            } else {
153                serde_json::Number::from_f64(*n)
154            };
155            if let Some(r) = r {
156                Ok(serde_json::Value::Number(r))
157            } else {
158                Err(format!("Could not convert {n} into a number"))
159            }
160        }
161        Value::EnumerationValue(e, v) => Ok(serde_json::Value::String(format!("{e}.{v}"))),
162        Value::String(shared_string) => Ok(serde_json::Value::String(shared_string.to_string())),
163        Value::Image(image) => {
164            if let Some(p) = image.path() {
165                Ok(serde_json::Value::String(format!("{}", p.to_string_lossy())))
166            } else {
167                Err("Cannot serialize an image without a path".into())
168            }
169        }
170        Value::Model(model_rc) => Ok(serde_json::Value::Array(
171            model_rc.iter().map(|v| v.to_json()).collect::<Result<Vec<_>, _>>()?,
172        )),
173        Value::Struct(s) => Ok(serde_json::Value::Object(
174            s.iter()
175                .map(|(k, v)| v.to_json().map(|v| (k.to_string(), v)))
176                .collect::<Result<serde_json::Map<_, _>, _>>()?,
177        )),
178        Value::Brush(brush) => match brush {
179            Brush::SolidColor(color) => Ok(serde_json::Value::String(color_to_string(color))),
180            Brush::LinearGradient(_) => Err("Cannot serialize a linear gradient".into()),
181            Brush::RadialGradient(_) => Err("Cannot serialize a radial gradient".into()),
182            _ => Err("Cannot serialize an unknown brush type".into()),
183        },
184        Value::PathData(_) => Err("Cannot serialize path data".into()),
185        Value::EasingCurve(_) => Err("Cannot serialize a easing curve".into()),
186        _ => Err("Cannot serialize an unknown value type".into()),
187    }
188}

190/// Write the `Value` out into a JSON string
191pub fn value_to_json_string(value: &Value) -> Result<String, String> {
192    Ok(value_to_json(value)?.to_string())
193}

195#[test]
196fn test_from_json() {
197    let v = value_from_json_str(&langtype::Type::Void, "null").unwrap();
198    assert_eq!(v, Value::Void);
199    let v = Value::from_json_str(&langtype::Type::Void, "null").unwrap();
200    assert_eq!(v, Value::Void);
201
202    let v = value_from_json_str(&langtype::Type::Float32, "42.0").unwrap();
203    assert_eq!(v, Value::Number(42.0));
204
205    let v = value_from_json_str(&langtype::Type::Int32, "23").unwrap();
206    assert_eq!(v, Value::Number(23.0));
207
208    let v = value_from_json_str(&langtype::Type::String, "\"a string with \\\\ escape\"").unwrap();
209    assert_eq!(v, Value::String("a string with \\ escape".into()));
210
211    let v = value_from_json_str(&langtype::Type::Color, "\"#0ab0cdff\"").unwrap();
212    assert_eq!(v, Value::Brush(Brush::SolidColor(Color::from_argb_u8(0xff, 0x0a, 0xb0, 0xcd))));
213    let v = value_from_json_str(&langtype::Type::Brush, "\"#0ab0cdff\"").unwrap();
214    assert_eq!(v, Value::Brush(Brush::SolidColor(Color::from_argb_u8(0xff, 0x0a, 0xb0, 0xcd))));
215}

217#[test]
218fn test_to_json() {
219    let v = value_to_json_string(&Value::Void).unwrap();
220    assert_eq!(v, "null".to_string());
221    let v = Value::Void.to_json_string().unwrap();
222    assert_eq!(v, "null".to_string());
223
224    let v = value_to_json_string(&Value::Number(23.0)).unwrap();
225    assert_eq!(v, "23".to_string());
226
227    let v = value_to_json_string(&Value::Number(4.2)).unwrap();
228    assert_eq!(v, "4.2".to_string());
229
230    let v = value_to_json_string(&Value::EnumerationValue("Foo".to_string(), "bar".to_string()))
231        .unwrap();
232    assert_eq!(v, "\"Foo.bar\"".to_string());
233
234    let v = value_to_json_string(&Value::String("Hello World with \\ escaped".into())).unwrap();
235    assert_eq!(v, "\"Hello World with \\\\ escaped\"".to_string());
236
237    // Image without path:
238    let buffer = i_slint_core::graphics::SharedPixelBuffer::new(2, 2);
239    assert!(value_to_json_string(&Value::Image(Image::from_rgb8(buffer))).is_err());
240
241    // Image with path
242    let path = std::path::PathBuf::from(env!("CARGO_MANIFEST_DIR"))
243        .join("../../logo/MadeWithSlint-logo-dark.png")
244        .canonicalize()
245        .unwrap();
246    let v = value_to_json_string(&Value::Image(Image::load_from_path(&path).unwrap())).unwrap();
247    // We are looking at the JSON string which needs to be escaped!
248    let path = path.to_string_lossy().replace("\\", "\\\\");
249    assert_eq!(v, format!("\"{path}\""));
250
251    let v = value_to_json_string(&Value::Bool(true)).unwrap();
252    assert_eq!(v, "true".to_string());
253
254    let v = value_to_json_string(&Value::Bool(false)).unwrap();
255    assert_eq!(v, "false".to_string());
256
257    let model: ModelRc<Value> = std::rc::Rc::new(i_slint_core::model::VecModel::from(vec![
258        Value::Bool(true),
259        Value::Bool(false),
260    ]))
261    .into();
262    let v = value_to_json_string(&Value::Model(model)).unwrap();
263    assert_eq!(v, "[true,false]".to_string());
264
265    let v = value_to_json_string(&Value::Struct(crate::Struct::from_iter([
266        ("kind".to_string(), Value::EnumerationValue("test".to_string(), "foo".to_string())),
267        ("is_bool".to_string(), Value::Bool(false)),
268        ("string-value".to_string(), Value::String("some string".into())),
269    ])))
270    .unwrap();
271    assert_eq!(
272        v,
273        "{\"is-bool\":false,\"kind\":\"test.foo\",\"string-value\":\"some string\"}".to_string()
274    );
275
276    let v = value_to_json_string(&Value::Brush(Brush::SolidColor(Color::from_argb_u8(
277        0xff, 0x0a, 0xb0, 0xcd,
278    ))))
279    .unwrap();
280    assert_eq!(v, "\"#0ab0cdff\"".to_string());
