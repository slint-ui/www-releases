94/// If the value contains a model set from [`slint_interpreter_value_new_model]` with the same vtable pointer,
95/// return the model that was set.
96/// Returns a null ptr otherwise
97#[unsafe(no_mangle)]
98pub extern "C" fn slint_interpreter_value_to_model(
99    val: &Value,
100    vtable: &ModelAdaptorVTable,
101) -> *const u8 {
102    if let Value::Model(m) = val {
103        if let Some(m) = m.as_any().downcast_ref::<ModelAdaptorWrapper>() {
104            if core::ptr::eq(m.0.get_vtable() as *const _, vtable as *const _) {
105                return m.0.as_ptr();
106            }
107        }
108    }
109    core::ptr::null()
110}

142/// `out` parameter and returns true; returns false if the value does not hold an extractable

199    }
200    fn as_struct_mut(&mut self) -> &mut Struct {
201        // Safety: there should be no way to construct a StructOpaque without it holding an actual Struct
202        unsafe { std::mem::transmute::<&mut StructOpaque, &mut Struct>(self) }
203    }
204}

263/// Advance the iterator and return the next value, or a null pointer
264#[unsafe(no_mangle)]
265pub unsafe extern "C" fn slint_interpreter_struct_iterator_next<'a>(
266    iter: &'a mut StructIteratorOpaque,
267    k: &mut Slice<'a, u8>,
268) -> *mut Value {
269    if let Some((str, val)) = (*(iter as *mut StructIteratorOpaque as *mut StructIterator)).next() {
270        *k = Slice::from_slice(str.as_bytes());
271        Box::into_raw(Box::new(val.clone()))
272    } else {
273        *k = Slice::default();
274        std::ptr::null_mut()
275    }
276}

278#[unsafe(no_mangle)]
279pub extern "C" fn slint_interpreter_struct_make_iter(
280    stru: &StructOpaque,
281) -> StructIteratorOpaque<'_> {
282    let ret_it: StructIterator = stru.as_struct().0.iter();
283    unsafe {
284        let mut r = std::mem::MaybeUninit::<StructIteratorOpaque>::uninit();
285        std::ptr::write(r.as_mut_ptr() as *mut StructIterator, ret_it);
286        r.assume_init()
287    }
288}

344/// Wrap the user_data provided by the native code and call the drop function on Drop.
345///
346/// Safety: user_data must be a pointer that can be destroyed by the drop_user_data function.
347/// callback must be a valid callback that initialize the `ret`
348pub struct CallbackUserData {
349    user_data: *mut c_void,
350    drop_user_data: Option<extern "C" fn(*mut c_void)>,
351    callback: extern "C" fn(user_data: *mut c_void, arg: Slice<Box<Value>>) -> Box<Value>,
352}
353
354impl Drop for CallbackUserData {
355    fn drop(&mut self) {
356        if let Some(x) = self.drop_user_data {
357            x(self.user_data)
358        }
359    }
360}
361
362impl CallbackUserData {
363    pub unsafe fn new(
364        user_data: *mut c_void,
365        drop_user_data: Option<extern "C" fn(*mut c_void)>,
366        callback: extern "C" fn(user_data: *mut c_void, arg: Slice<Box<Value>>) -> Box<Value>,
367    ) -> Self {
368        Self { user_data, drop_user_data, callback }
369    }
370
371    pub fn call(&self, args: &[Value]) -> Value {
372        let args = args.iter().map(|v| v.clone().into()).collect::<Vec<_>>();
373        (self.callback)(self.user_data, Slice::from_slice(args.as_ref())).as_ref().clone()
374    }
375}

377/// Set a handler for the callback.
378/// The `callback` function must initialize the `ret` (the `ret` passed to the callback is initialized and is assumed initialized after the function)
379#[unsafe(no_mangle)]
380pub unsafe extern "C" fn slint_interpreter_component_instance_set_callback(
381    inst: &ErasedItemTreeBox,
382    name: Slice<u8>,
383    callback: extern "C" fn(user_data: *mut c_void, arg: Slice<Box<Value>>) -> Box<Value>,
384    user_data: *mut c_void,
385    drop_user_data: Option<extern "C" fn(*mut c_void)>,
386) -> bool {
387    let ud = CallbackUserData::new(user_data, drop_user_data, callback);
388
389    generativity::make_guard!(guard);
390    let comp = inst.unerase(guard);
391    comp.description()
392        .set_callback_handler(
393            comp.borrow(),
394            &normalize_identifier(std::str::from_utf8(&name).unwrap()),
395            Box::new(move |args| ud.call(args)),
396        )
397        .is_ok()
398}

468/// Invoke a global callback or function. Returns raw boxed value on success; nullptr otherwise.
469#[unsafe(no_mangle)]
470pub unsafe extern "C" fn slint_interpreter_component_instance_invoke_global(
471    inst: &ErasedItemTreeBox,
472    global: Slice<u8>,
473    callable_name: Slice<u8>,
474    args: Slice<Box<Value>>,
475) -> *mut Value {
476    let args = args.iter().map(|vb| vb.as_ref().clone()).collect::<Vec<_>>();
477    generativity::make_guard!(guard);
478    let comp = inst.unerase(guard);
479    let callable_name = std::str::from_utf8(&callable_name).unwrap();
480    match comp
481        .description()
482        .get_global(comp.borrow(), &normalize_identifier(std::str::from_utf8(&global).unwrap()))
483        .and_then(|g| {
484            if matches!(
485                comp.description()
486                    .original
487                    .root_element
488                    .borrow()
489                    .lookup_property(callable_name)
490                    .property_type,
491                i_slint_compiler::langtype::Type::Function { .. }
492            ) {
493                g.as_ref().eval_function(
494                    &normalize_identifier(callable_name),
495                    args.as_slice().iter().cloned().collect(),
496                )
497            } else {
498                g.as_ref().invoke_callback(&normalize_identifier(callable_name), args.as_slice())
499            }
500        }) {
501        Ok(val) => Box::into_raw(Box::new(val)),
502        Err(_) => std::ptr::null_mut(),
503    }
504}

566    pub set_row_data: extern "C" fn(VRef<ModelAdaptorVTable>, row: usize, value: Box<Value>),
567    pub get_notify: extern "C" fn(VRef<'_, ModelAdaptorVTable>) -> &ModelNotifyOpaque,

571struct ModelAdaptorWrapper(vtable::VBox<ModelAdaptorVTable>);
572impl Model for ModelAdaptorWrapper {
573    type Data = Value;
574
575    fn row_count(&self) -> usize {
576        self.0.row_count()
577    }
578
579    fn row_data(&self, row: usize) -> Option<Value> {
580        let val_ptr = unsafe { self.0.row_data(row) };
581        if val_ptr.is_null() {
582            None
583        } else {
584            Some(*unsafe { Box::from_raw(val_ptr) })
585        }
586    }
587
588    fn model_tracker(&self) -> &dyn i_slint_core::model::ModelTracker {
589        self.0.get_notify().as_model_notify()
590    }
591
592    fn set_row_data(&self, row: usize, data: Value) {
593        let val = Box::new(data);
594        self.0.set_row_data(row, val);
595    }
596
597    fn as_any(&self) -> &dyn core::any::Any {
598        self
599    }
600}

673/// Diagnostic describes the aspects of either a warning or an error, along

776    compiler: &ComponentCompilerOpaque,
777    out_diags: &mut SharedVector<Diagnostic>,
778) {
779    #[allow(deprecated)]
780    out_diags.extend(compiler.as_component_compiler().diagnostics.iter().map(|diagnostic| {
781        let (line, column) = diagnostic.line_column();
782        Diagnostic {
783            message: diagnostic.message().into(),
784            source_file: diagnostic
785                .source_file()
786                .and_then(|path| path.to_str())
787                .map_or_else(Default::default, |str| str.into()),
788            line,
789            column,
790            level: match diagnostic.level() {
791                i_slint_compiler::diagnostics::DiagnosticLevel::Error => DiagnosticLevel::Error,
792                i_slint_compiler::diagnostics::DiagnosticLevel::Warning => DiagnosticLevel::Warning,
793                _ => DiagnosticLevel::Warning,
794            },
795        }
796    }));
797}

799#[unsafe(no_mangle)]
800pub unsafe extern "C" fn slint_interpreter_component_compiler_build_from_source(
801    compiler: &mut ComponentCompilerOpaque,
802    source_code: Slice<u8>,
803    path: Slice<u8>,
804    component_definition_ptr: *mut ComponentDefinitionOpaque,
805) -> bool {
806    match spin_on::spin_on(compiler.as_component_compiler_mut().build_from_source(
807        std::str::from_utf8(&source_code).unwrap().to_string(),
808        std::str::from_utf8(&path).unwrap().to_string().into(),
809    )) {
810        Some(definition) => {

812            true
813        }
814        None => false,
815    }
816}
