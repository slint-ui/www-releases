28pub trait ErasedPropertyInfo {
29    fn get(&self, item: Pin<ItemRef>) -> Value;
30    fn set(
31        &self,
32        item: Pin<ItemRef>,
33        value: Value,
34        animation: Option<PropertyAnimation>,
35    ) -> Result<(), ()>;
36    fn set_binding(
37        &self,
38        item: Pin<ItemRef>,
39        binding: Box<dyn Fn() -> Value>,
40        animation: AnimatedBindingKind,
41    );
42    fn offset(&self) -> usize;
43
44    /// Safety: Property2 must be a (pinned) pointer to a `Property<T>`
45    /// where T is the same T as the one represented by this property.
46    unsafe fn link_two_ways(&self, item: Pin<ItemRef>, property2: *const ());
47}

49impl<Item: vtable::HasStaticVTable<corelib::items::ItemVTable>> ErasedPropertyInfo
50    for &'static dyn corelib::rtti::PropertyInfo<Item, Value>
51{
52    fn get(&self, item: Pin<ItemRef>) -> Value {
53        (*self).get(ItemRef::downcast_pin(item).unwrap()).unwrap()
54    }
55    fn set(
56        &self,
57        item: Pin<ItemRef>,
58        value: Value,
59        animation: Option<PropertyAnimation>,
60    ) -> Result<(), ()> {
61        (*self).set(ItemRef::downcast_pin(item).unwrap(), value, animation)
62    }
63    fn set_binding(
64        &self,
65        item: Pin<ItemRef>,
66        binding: Box<dyn Fn() -> Value>,
67        animation: AnimatedBindingKind,
68    ) {
69        (*self).set_binding(ItemRef::downcast_pin(item).unwrap(), binding, animation).unwrap();
70    }
71    fn offset(&self) -> usize {
72        (*self).offset()
73    }
74    unsafe fn link_two_ways(&self, item: Pin<ItemRef>, property2: *const ()) {
75        // Safety: ErasedPropertyInfo::link_two_ways and PropertyInfo::link_two_ways have the same safety requirement
76        (*self).link_two_ways(ItemRef::downcast_pin(item).unwrap(), property2)
77    }
78}

138/// Evaluate an expression and return a Value as the result of this expression
139pub fn eval_expression(expression: &Expression, local_context: &mut EvalLocalContext) -> Value {
140    if let Some(r) = &local_context.return_value {
141        return r.clone();
142    }
143    match expression {
144        Expression::Invalid => panic!("invalid expression while evaluating"),
145        Expression::Uncompiled(_) => panic!("uncompiled expression while evaluating"),
146        Expression::StringLiteral(s) => Value::String(s.as_str().into()),
147        Expression::NumberLiteral(n, unit) => Value::Number(unit.normalize(*n)),
148        Expression::BoolLiteral(b) => Value::Bool(*b),
149        Expression::ElementReference(_) => todo!("Element references are only supported in the context of built-in function calls at the moment"),
150        Expression::PropertyReference(nr) => {
151            load_property_helper(&ComponentInstance::InstanceRef(local_context.component_instance), &nr.element(), nr.name()).unwrap()
152        }
153        Expression::RepeaterIndexReference { element } => load_property_helper(&ComponentInstance::InstanceRef(local_context.component_instance),
154            &element.upgrade().unwrap().borrow().base_type.as_component().root_element,
155            crate::dynamic_item_tree::SPECIAL_PROPERTY_INDEX,
156        )
157        .unwrap(),
158        Expression::RepeaterModelReference { element } => {
159            let value = load_property_helper(&ComponentInstance::InstanceRef(local_context.component_instance),
160                    &element.upgrade().unwrap().borrow().base_type.as_component().root_element,
161                    crate::dynamic_item_tree::SPECIAL_PROPERTY_MODEL_DATA,
162                )
163                .unwrap();
164            if matches!(value, Value::Void) {
165                // Uninitialized model data (because the model returned None) should still be initialized to the default value of the type
166                default_value_for_type(&expression.ty())
167            } else {
168                value
169            }
170
171        },
172        Expression::FunctionParameterReference { index, .. } => {
173            local_context.function_arguments[*index].clone()
174        }
175        Expression::StructFieldAccess { base, name } => {
176            if let Value::Struct(o) = eval_expression(base, local_context) {
177                o.get_field(name).cloned().unwrap_or(Value::Void)
178            } else {
179                Value::Void
180            }
181        }
182        Expression::ArrayIndex { array, index } => {
183            let array = eval_expression(array, local_context);
184            let index = eval_expression(index, local_context);
185            match (array, index) {
186                (Value::Model(model), Value::Number(index)) => {
187                    model.row_data_tracked(index as isize as usize).unwrap_or_else(|| default_value_for_type(&expression.ty()))
188                }
189                _ => {
190                    Value::Void
191                }
192            }
193        }
194        Expression::Cast { from, to } => {
195            let v = eval_expression(from, local_context);
196            match (v, to) {
197                (Value::Number(n), Type::Int32) => Value::Number(n.trunc()),
198                (Value::Number(n), Type::String) => {
199                    Value::String(i_slint_core::string::shared_string_from_number(n))
200                }
201                (Value::Number(n), Type::Color) => Color::from_argb_encoded(n as u32).into(),
202                (Value::Brush(brush), Type::Color) => brush.color().into(),
203                (v, _) => v,
204            }
205        }
206        Expression::CodeBlock(sub) => {
207            let mut v = Value::Void;
208            for e in sub {
209                v = eval_expression(e, local_context);
210                if let Some(r) = &local_context.return_value {
211                    return r.clone();
212                }
213            }
214            v
215        }
216        Expression::FunctionCall { function, arguments, source_location } => match &function {
217            Callable::Function(nr) => {
218                let is_item_member = nr.element().borrow().native_class().is_some_and(|n| n.properties.contains_key(nr.name()));
219                if is_item_member {
220                    call_item_member_function(nr, local_context)
221                } else {
222                    let args = arguments.iter().map(|e| eval_expression(e, local_context)).collect::<Vec<_>>();
223                    call_function(&ComponentInstance::InstanceRef(local_context.component_instance), &nr.element(), nr.name(), args).unwrap()
224                }
225            }
226            Callable::Callback(nr) => {
227                let args = arguments.iter().map(|e| eval_expression(e, local_context)).collect::<Vec<_>>();
228                invoke_callback(&ComponentInstance::InstanceRef(local_context.component_instance), &nr.element(), nr.name(), &args).unwrap()
229            }
230            Callable::Builtin(f) => call_builtin_function(f.clone(), arguments, local_context, source_location),
231        }
232        Expression::SelfAssignment { lhs, rhs, op, .. } => {
233            let rhs = eval_expression(rhs, local_context);
234            eval_assignment(lhs, *op, rhs, local_context);
235            Value::Void
236        }
237        Expression::BinaryExpression { lhs, rhs, op } => {
238            let lhs = eval_expression(lhs, local_context);
239            let rhs = eval_expression(rhs, local_context);
240
241            match (op, lhs, rhs) {
242                ('+', Value::String(mut a), Value::String(b)) => { a.push_str(b.as_str()); Value::String(a) },
243                ('+', Value::Number(a), Value::Number(b)) => Value::Number(a + b),
244                ('+', a @ Value::Struct(_), b @ Value::Struct(_)) => {
245                    let a : Option<corelib::layout::LayoutInfo> = a.try_into().ok();
246                    let b : Option<corelib::layout::LayoutInfo> = b.try_into().ok();
247                    if let (Some(a), Some(b)) = (a, b) {
248                        a.merge(&b).into()
249                    } else {
250                        panic!("unsupported {a:?} {op} {b:?}");
251                    }
252                }
253                ('-', Value::Number(a), Value::Number(b)) => Value::Number(a - b),
254                ('/', Value::Number(a), Value::Number(b)) => Value::Number(a / b),
255                ('*', Value::Number(a), Value::Number(b)) => Value::Number(a * b),
256                ('<', Value::Number(a), Value::Number(b)) => Value::Bool(a < b),
257                ('>', Value::Number(a), Value::Number(b)) => Value::Bool(a > b),
258                ('≤', Value::Number(a), Value::Number(b)) => Value::Bool(a <= b),
259                ('≥', Value::Number(a), Value::Number(b)) => Value::Bool(a >= b),
260                ('<', Value::String(a), Value::String(b)) => Value::Bool(a < b),
261                ('>', Value::String(a), Value::String(b)) => Value::Bool(a > b),
262                ('≤', Value::String(a), Value::String(b)) => Value::Bool(a <= b),
263                ('≥', Value::String(a), Value::String(b)) => Value::Bool(a >= b),
264                ('=', a, b) => Value::Bool(a == b),
265                ('!', a, b) => Value::Bool(a != b),
266                ('&', Value::Bool(a), Value::Bool(b)) => Value::Bool(a && b),
267                ('|', Value::Bool(a), Value::Bool(b)) => Value::Bool(a || b),
268                (op, lhs, rhs) => panic!("unsupported {lhs:?} {op} {rhs:?}"),
269            }
270        }
271        Expression::UnaryOp { sub, op } => {
272            let sub = eval_expression(sub, local_context);
273            match (sub, op) {
274                (Value::Number(a), '+') => Value::Number(a),
275                (Value::Number(a), '-') => Value::Number(-a),
276                (Value::Bool(a), '!') => Value::Bool(!a),
277                (sub, op) => panic!("unsupported {op} {sub:?}"),
278            }
279        }
280        Expression::ImageReference{ resource_ref, nine_slice, .. } => {
281            let mut image = match resource_ref {
282                i_slint_compiler::expression_tree::ImageReference::None => {
283                    Ok(Default::default())
284                }
285                i_slint_compiler::expression_tree::ImageReference::AbsolutePath(path) => {
286                    let path = std::path::Path::new(path);
287                    if path.starts_with("builtin:/") {
288                        i_slint_compiler::fileaccess::load_file(path).and_then(|virtual_file| virtual_file.builtin_contents).map(|virtual_file| {
289                            let extension = path.extension().unwrap().to_str().unwrap();
290                            corelib::graphics::load_image_from_embedded_data(
291                                corelib::slice::Slice::from_slice(virtual_file),
292                                corelib::slice::Slice::from_slice(extension.as_bytes())
293                            )
294                        }).ok_or_else(Default::default)
295                    } else {
296                        corelib::graphics::Image::load_from_path(path)
297                    }
298                }
299                i_slint_compiler::expression_tree::ImageReference::EmbeddedData { .. } => {
300                    todo!()
301                }
302                i_slint_compiler::expression_tree::ImageReference::EmbeddedTexture { .. } => {
303                    todo!()
304                }
305            }.unwrap_or_else(|_| {
306                eprintln!("Could not load image {resource_ref:?}" );
307                Default::default()
308            });
309            if let Some(n) = nine_slice {
310                image.set_nine_slice_edges(n[0], n[1], n[2], n[3]);
311            }
312            Value::Image(image)
313        }
314        Expression::Condition { condition, true_expr, false_expr } => {
315            match eval_expression(condition, local_context).try_into()
316                as Result<bool, _>
317            {
318                Ok(true) => eval_expression(true_expr, local_context),
319                Ok(false) => eval_expression(false_expr, local_context),
320                _ => local_context.return_value.clone().expect("conditional expression did not evaluate to boolean"),
321            }
322        }
323        Expression::Array { values, .. } => Value::Model(
324            ModelRc::new(corelib::model::SharedVectorModel::from(
325                values.iter().map(|e| eval_expression(e, local_context)).collect::<SharedVector<_>>()
326            )
327        )),
328        Expression::Struct { values, .. } => Value::Struct(
329            values
330                .iter()
331                .map(|(k, v)| (k.to_string(), eval_expression(v, local_context)))
332                .collect(),
333        ),
334        Expression::PathData(data)  => {
335            Value::PathData(convert_path(data, local_context))
336        }
337        Expression::StoreLocalVariable { name, value } => {
338            let value = eval_expression(value, local_context);
339            local_context.local_variables.insert(name.clone(), value);
340            Value::Void
341        }
342        Expression::ReadLocalVariable { name, .. } => {
343            local_context.local_variables.get(name).unwrap().clone()
344        }
345        Expression::EasingCurve(curve) => Value::EasingCurve(match curve {
346            EasingCurve::Linear => corelib::animations::EasingCurve::Linear,
347            EasingCurve::EaseInElastic => corelib::animations::EasingCurve::EaseInElastic,
348            EasingCurve::EaseOutElastic => corelib::animations::EasingCurve::EaseOutElastic,
349            EasingCurve::EaseInOutElastic => corelib::animations::EasingCurve::EaseInOutElastic,
350            EasingCurve::EaseInBounce => corelib::animations::EasingCurve::EaseInBounce,
351            EasingCurve::EaseOutBounce => corelib::animations::EasingCurve::EaseOutBounce,
352            EasingCurve::EaseInOutBounce => corelib::animations::EasingCurve::EaseInOutBounce,
353            EasingCurve::CubicBezier(a, b, c, d) => {
354                corelib::animations::EasingCurve::CubicBezier([*a, *b, *c, *d])
355            }
356        }),
357        Expression::LinearGradient{angle, stops} => {
358            let angle = eval_expression(angle, local_context);
359            Value::Brush(Brush::LinearGradient(LinearGradientBrush::new(angle.try_into().unwrap(), stops.iter().map(|(color, stop)| {
360                let color = eval_expression(color, local_context).try_into().unwrap();
361                let position = eval_expression(stop, local_context).try_into().unwrap();
362                GradientStop{ color, position }
363            }))))
364        }
365        Expression::RadialGradient{stops} => {
366            Value::Brush(Brush::RadialGradient(RadialGradientBrush::new_circle(stops.iter().map(|(color, stop)| {
367                let color = eval_expression(color, local_context).try_into().unwrap();
368                let position = eval_expression(stop, local_context).try_into().unwrap();
369                GradientStop{ color, position }
370            }))))
371        }
372        Expression::EnumerationValue(value) => {
373            Value::EnumerationValue(value.enumeration.name.to_string(), value.to_string())
374        }
375        Expression::ReturnStatement(x) => {
376            let val = x.as_ref().map_or(Value::Void, |x| eval_expression(x, local_context));
377            if local_context.return_value.is_none() {
378                local_context.return_value = Some(val);
379            }
380            local_context.return_value.clone().unwrap()
381        }
382        Expression::LayoutCacheAccess { layout_cache_prop, index, repeater_index } => {
383            let cache = load_property_helper(&ComponentInstance::InstanceRef(local_context.component_instance), &layout_cache_prop.element(), layout_cache_prop.name()).unwrap();
384            if let Value::LayoutCache(cache) = cache {
385                if let Some(ri) = repeater_index {
386                    let offset : usize = eval_expression(ri, local_context).try_into().unwrap();
387                    Value::Number(cache.get((cache[*index] as usize) + offset * 2).copied().unwrap_or(0.).into())
388                } else {
389                    Value::Number(cache[*index].into())
390                }
391            } else {
392                panic!("invalid layout cache")
393            }
394        }
395        Expression::ComputeLayoutInfo(lay, o) => crate::eval_layout::compute_layout_info(lay, *o, local_context),
396        Expression::SolveLayout(lay, o) => crate::eval_layout::solve_layout(lay, *o, local_context),
397        Expression::MinMax { ty: _, op, lhs, rhs } => {
398            let Value::Number(lhs) = eval_expression(lhs, local_context) else {
399                return local_context.return_value.clone().expect("minmax lhs expression did not evaluate to number");
400            };
401            let Value::Number(rhs) =  eval_expression(rhs, local_context) else {
402                return local_context.return_value.clone().expect("minmax rhs expression did not evaluate to number");
403            };
404            match op {
405                MinMaxOp::Min => Value::Number(lhs.min(rhs)),
406                MinMaxOp::Max => Value::Number(lhs.max(rhs)),
407            }
408        }
409        Expression::EmptyComponentFactory => Value::ComponentFactory(Default::default()),
410        Expression::DebugHook { expression, .. } => eval_expression(expression, local_context),
411    }
412}

414fn call_builtin_function(
415    f: BuiltinFunction,
416    arguments: &[Expression],
417    local_context: &mut EvalLocalContext,
418    source_location: &Option<i_slint_compiler::diagnostics::SourceLocation>,
419) -> Value {
420    match f {
421        BuiltinFunction::GetWindowScaleFactor => Value::Number(
422            local_context.component_instance.access_window(|window| window.scale_factor()) as _,
423        ),
424        BuiltinFunction::GetWindowDefaultFontSize => {
425            Value::Number(local_context.component_instance.access_window(|window| {
426                window.window_item().unwrap().as_pin_ref().default_font_size().get()
427            }) as _)
428        }
429        BuiltinFunction::AnimationTick => {
430            Value::Number(i_slint_core::animations::animation_tick() as f64)
431        }
432        BuiltinFunction::Debug => {
433            let to_print: SharedString =
434                eval_expression(&arguments[0], local_context).try_into().unwrap();
435            local_context.component_instance.description.debug_handler.borrow()(
436                source_location.as_ref(),
437                &to_print,
438            );
439            Value::Void
440        }
441        BuiltinFunction::Mod => {
442            let mut to_num = |e| -> f64 { eval_expression(e, local_context).try_into().unwrap() };
443            Value::Number(to_num(&arguments[0]).rem_euclid(to_num(&arguments[1])))
444        }
445        BuiltinFunction::Round => {
446            let x: f64 = eval_expression(&arguments[0], local_context).try_into().unwrap();
447            Value::Number(x.round())
448        }
449        BuiltinFunction::Ceil => {
450            let x: f64 = eval_expression(&arguments[0], local_context).try_into().unwrap();
451            Value::Number(x.ceil())
452        }
453        BuiltinFunction::Floor => {
454            let x: f64 = eval_expression(&arguments[0], local_context).try_into().unwrap();
455            Value::Number(x.floor())
456        }
457        BuiltinFunction::Sqrt => {
458            let x: f64 = eval_expression(&arguments[0], local_context).try_into().unwrap();
459            Value::Number(x.sqrt())
460        }
461        BuiltinFunction::Abs => {
462            let x: f64 = eval_expression(&arguments[0], local_context).try_into().unwrap();
463            Value::Number(x.abs())
464        }
465        BuiltinFunction::Sin => {
466            let x: f64 = eval_expression(&arguments[0], local_context).try_into().unwrap();
467            Value::Number(x.to_radians().sin())
468        }
469        BuiltinFunction::Cos => {
470            let x: f64 = eval_expression(&arguments[0], local_context).try_into().unwrap();
471            Value::Number(x.to_radians().cos())
472        }
473        BuiltinFunction::Tan => {
474            let x: f64 = eval_expression(&arguments[0], local_context).try_into().unwrap();
475            Value::Number(x.to_radians().tan())
476        }
477        BuiltinFunction::ASin => {
478            let x: f64 = eval_expression(&arguments[0], local_context).try_into().unwrap();
479            Value::Number(x.asin().to_degrees())
480        }
481        BuiltinFunction::ACos => {
482            let x: f64 = eval_expression(&arguments[0], local_context).try_into().unwrap();
483            Value::Number(x.acos().to_degrees())
484        }
485        BuiltinFunction::ATan => {
486            let x: f64 = eval_expression(&arguments[0], local_context).try_into().unwrap();
487            Value::Number(x.atan().to_degrees())
488        }
489        BuiltinFunction::ATan2 => {
490            let x: f64 = eval_expression(&arguments[0], local_context).try_into().unwrap();
491            let y: f64 = eval_expression(&arguments[1], local_context).try_into().unwrap();
492            Value::Number(x.atan2(y).to_degrees())
493        }
494        BuiltinFunction::Log => {
495            let x: f64 = eval_expression(&arguments[0], local_context).try_into().unwrap();
496            let y: f64 = eval_expression(&arguments[1], local_context).try_into().unwrap();
497            Value::Number(x.log(y))
498        }
499        BuiltinFunction::Ln => {
500            let x: f64 = eval_expression(&arguments[0], local_context).try_into().unwrap();
501            Value::Number(x.ln())
502        }
503        BuiltinFunction::Pow => {
504            let x: f64 = eval_expression(&arguments[0], local_context).try_into().unwrap();
505            let y: f64 = eval_expression(&arguments[1], local_context).try_into().unwrap();
506            Value::Number(x.powf(y))
507        }
508        BuiltinFunction::Exp => {
509            let x: f64 = eval_expression(&arguments[0], local_context).try_into().unwrap();
510            Value::Number(x.exp())
511        }
512        BuiltinFunction::ToFixed => {
513            let n: f64 = eval_expression(&arguments[0], local_context).try_into().unwrap();
514            let digits: i32 = eval_expression(&arguments[1], local_context).try_into().unwrap();
515            let digits: usize = digits.max(0) as usize;
516            Value::String(i_slint_core::string::shared_string_from_number_fixed(n, digits))
517        }
518        BuiltinFunction::ToPrecision => {
519            let n: f64 = eval_expression(&arguments[0], local_context).try_into().unwrap();
520            let precision: i32 = eval_expression(&arguments[1], local_context).try_into().unwrap();
521            let precision: usize = precision.max(0) as usize;

1359fn call_item_member_function(nr: &NamedReference, local_context: &mut EvalLocalContext) -> Value {
1360    let component = local_context.component_instance;
1361    let elem = nr.element();
1362    let name = nr.name().as_str();
1363    generativity::make_guard!(guard);
1364    let enclosing_component = enclosing_component_for_element(&elem, component, guard);
1365    let description = enclosing_component.description;
1366    let item_info = &description.items[elem.borrow().id.as_str()];
1367    let item_ref = unsafe { item_info.item_from_item_tree(enclosing_component.as_ptr()) };
1368
1369    let item_comp = enclosing_component.self_weak().get().unwrap().upgrade().unwrap();
1370    let item_rc =
1371        corelib::items::ItemRc::new(vtable::VRc::into_dyn(item_comp), item_info.item_index());
1372
1373    let window_adapter = component.window_adapter();
1374
1375    // TODO: Make this generic through RTTI
1376    if let Some(textinput) = ItemRef::downcast_pin::<corelib::items::TextInput>(item_ref) {
1377        match name {
1378            "select-all" => textinput.select_all(&window_adapter, &item_rc),
1379            "clear-selection" => textinput.clear_selection(&window_adapter, &item_rc),
1380            "cut" => textinput.cut(&window_adapter, &item_rc),
1381            "copy" => textinput.copy(&window_adapter, &item_rc),
1382            "paste" => textinput.paste(&window_adapter, &item_rc),
1383            _ => panic!("internal: Unknown member function {name} called on TextInput"),
1384        }
1385    } else if let Some(s) = ItemRef::downcast_pin::<corelib::items::SwipeGestureHandler>(item_ref) {
1386        match name {
1387            "cancel" => s.cancel(&window_adapter, &item_rc),
1388            _ => panic!("internal: Unknown member function {name} called on SwipeGestureHandler"),
1389        }
1390    } else if let Some(s) = ItemRef::downcast_pin::<corelib::items::ContextMenu>(item_ref) {
1391        match name {
1392            "close" => s.close(&window_adapter, &item_rc),
1393            "is-open" => return Value::Bool(s.is_open(&window_adapter, &item_rc)),
1394            _ => {
1395                panic!("internal: Unknown member function {name} called on ContextMenu")
1396            }
1397        }
1398    } else {
1399        panic!(
1400            "internal error: member function {name} called on element that doesn't have it: {}",
1401            elem.borrow().original_name()
1402        )
1403    }
1404
1405    Value::Void
1406}

1408fn eval_assignment(lhs: &Expression, op: char, rhs: Value, local_context: &mut EvalLocalContext) {
1409    let eval = |lhs| match (lhs, &rhs, op) {
1410        (Value::String(ref mut a), Value::String(b), '+') => {
1411            a.push_str(b.as_str());
1412            Value::String(a.clone())
1413        }
1414        (Value::Number(a), Value::Number(b), '+') => Value::Number(a + b),
1415        (Value::Number(a), Value::Number(b), '-') => Value::Number(a - b),
1416        (Value::Number(a), Value::Number(b), '/') => Value::Number(a / b),
1417        (Value::Number(a), Value::Number(b), '*') => Value::Number(a * b),
1418        (lhs, rhs, op) => panic!("unsupported {lhs:?} {op} {rhs:?}"),
1419    };
1420    match lhs {
1421        Expression::PropertyReference(nr) => {
1422            let element = nr.element();
1423            generativity::make_guard!(guard);
1424            let enclosing_component = enclosing_component_instance_for_element(
1425                &element,
1426                &ComponentInstance::InstanceRef(local_context.component_instance),
1427                guard,
1428            );
1429
1430            match enclosing_component {
1431                ComponentInstance::InstanceRef(enclosing_component) => {
1432                    if op == '=' {
1433                        store_property(enclosing_component, &element, nr.name(), rhs).unwrap();
1434                        return;
1435                    }
1436
1437                    let component = element.borrow().enclosing_component.upgrade().unwrap();
1438                    if element.borrow().id == component.root_element.borrow().id {
1439                        if let Some(x) =
1440                            enclosing_component.description.custom_properties.get(nr.name())
1441                        {
1442                            unsafe {
1443                                let p = Pin::new_unchecked(
1444                                    &*enclosing_component.as_ptr().add(x.offset),
1445                                );
1446                                x.prop.set(p, eval(x.prop.get(p).unwrap()), None).unwrap();
1447                            }
1448                            return;
1449                        }
1450                    };
1451                    let item_info =
1452                        &enclosing_component.description.items[element.borrow().id.as_str()];
1453                    let item =
1454                        unsafe { item_info.item_from_item_tree(enclosing_component.as_ptr()) };
1455                    let p = &item_info.rtti.properties[nr.name().as_str()];
1456                    p.set(item, eval(p.get(item)), None).unwrap();
1457                }
1458                ComponentInstance::GlobalComponent(global) => {
1459                    let val = if op == '=' {
1460                        rhs
1461                    } else {
1462                        eval(global.as_ref().get_property(nr.name()).unwrap())
1463                    };
1464                    global.as_ref().set_property(nr.name(), val).unwrap();
1465                }
1466            }
1467        }
1468        Expression::StructFieldAccess { base, name } => {
1469            if let Value::Struct(mut o) = eval_expression(base, local_context) {
1470                let mut r = o.get_field(name).unwrap().clone();
1471                r = if op == '=' { rhs } else { eval(std::mem::take(&mut r)) };
1472                o.set_field(name.to_string(), r);
1473                eval_assignment(base, '=', Value::Struct(o), local_context)
1474            }
1475        }
1476        Expression::RepeaterModelReference { element } => {
1477            let element = element.upgrade().unwrap();
1478            let component_instance = local_context.component_instance;
1479            generativity::make_guard!(g1);
1480            let enclosing_component =
1481                enclosing_component_for_element(&element, component_instance, g1);
1482            // we need a 'static Repeater component in order to call model_set_row_data, so get it.
1483            // Safety: This is the only 'static Id in scope.
1484            let static_guard =
1485                unsafe { generativity::Guard::new(generativity::Id::<'static>::new()) };
1486            let repeater = crate::dynamic_item_tree::get_repeater_by_name(
1487                enclosing_component,
1488                element.borrow().id.as_str(),
1489                static_guard,
1490            );
1491            repeater.0.model_set_row_data(
1492                eval_expression(
1493                    &Expression::RepeaterIndexReference { element: Rc::downgrade(&element) },
1494                    local_context,
1495                )
1496                .try_into()
1497                .unwrap(),
1498                if op == '=' {
1499                    rhs
1500                } else {
1501                    eval(eval_expression(
1502                        &Expression::RepeaterModelReference { element: Rc::downgrade(&element) },
1503                        local_context,
1504                    ))
1505                },
1506            )
1507        }
1508        Expression::ArrayIndex { array, index } => {
1509            let array = eval_expression(array, local_context);
1510            let index = eval_expression(index, local_context);
1511            match (array, index) {
1512                (Value::Model(model), Value::Number(index)) => {
1513                    if index >= 0. && (index as usize) < model.row_count() {
1514                        let index = index as usize;
1515                        if op == '=' {
1516                            model.set_row_data(index, rhs);
1517                        } else {
1518                            model.set_row_data(
1519                                index,
1520                                eval(
1521                                    model
1522                                        .row_data(index)
1523                                        .unwrap_or_else(|| default_value_for_type(&lhs.ty())),
1524                                ),
1525                            );
1526                        }
1527                    }
1528                }
1529                _ => {
1530                    eprintln!("Attempting to write into an array that cannot be written");
1531                }
1532            }
1533        }
1534        _ => panic!("typechecking should make sure this was a PropertyReference"),
1535    }
1536}

1542fn load_property_helper(
1543    component_instance: &ComponentInstance,
1544    element: &ElementRc,
1545    name: &str,
1546) -> Result<Value, ()> {
1547    generativity::make_guard!(guard);
1548    match enclosing_component_instance_for_element(element, component_instance, guard) {
1549        ComponentInstance::InstanceRef(enclosing_component) => {

1552            {
1553                if let Some(x) = enclosing_component.description.custom_properties.get(name) {
1554                    return unsafe {
1555                        x.prop.get(Pin::new_unchecked(&*enclosing_component.as_ptr().add(x.offset)))
1556                    };
1557                } else if enclosing_component.description.original.is_global() {
1558                    return Err(());
1559                }

1562                .description
1563                .items

1567            let item = unsafe { item_info.item_from_item_tree(enclosing_component.as_ptr()) };
1568            Ok(item_info.rtti.properties.get(name).ok_or(())?.get(item))
1569        }
1570        ComponentInstance::GlobalComponent(glob) => glob.as_ref().get_property(name),
1571    }

1574pub fn store_property(
1575    component_instance: InstanceRef,
1576    element: &ElementRc,
1577    name: &str,
1578    value: Value,
1579) -> Result<(), SetPropertyError> {
1580    generativity::make_guard!(guard);
1581    match enclosing_component_instance_for_element(
1582        element,
1583        &ComponentInstance::InstanceRef(component_instance),
1584        guard,
1585    ) {
1586        ComponentInstance::InstanceRef(enclosing_component) => {
1587            let maybe_animation = match element.borrow().bindings.get(name) {
1588                Some(b) => crate::dynamic_item_tree::animation_for_property(
1589                    enclosing_component,
1590                    &b.borrow().animation,
1591                ),
1592                None => {
1593                    crate::dynamic_item_tree::animation_for_property(enclosing_component, &None)
1594                }
1595            };
1596
1597            let component = element.borrow().enclosing_component.upgrade().unwrap();
1598            if element.borrow().id == component.root_element.borrow().id {
1599                if let Some(x) = enclosing_component.description.custom_properties.get(name) {
1600                    if let Some(orig_decl) = enclosing_component
1601                        .description
1602                        .original
1603                        .root_element
1604                        .borrow()
1605                        .property_declarations
1606                        .get(name)
1607                    {
1608                        // Do an extra type checking because PropertyInfo::set won't do it for custom structures or array
1609                        if !check_value_type(&value, &orig_decl.property_type) {
1610                            return Err(SetPropertyError::WrongType);
1611                        }
1612                    }
1613                    unsafe {
1614                        let p = Pin::new_unchecked(&*enclosing_component.as_ptr().add(x.offset));
1615                        return x
1616                            .prop
1617                            .set(p, value, maybe_animation.as_animation())
1618                            .map_err(|()| SetPropertyError::WrongType);
1619                    }
1620                } else if enclosing_component.description.original.is_global() {
1621                    return Err(SetPropertyError::NoSuchProperty);
1622                }
1623            };
1624            let item_info = &enclosing_component.description.items[element.borrow().id.as_str()];
1625            let item = unsafe { item_info.item_from_item_tree(enclosing_component.as_ptr()) };
1626            let p = &item_info.rtti.properties.get(name).ok_or(SetPropertyError::NoSuchProperty)?;
1627            p.set(item, value, maybe_animation.as_animation())
1628                .map_err(|()| SetPropertyError::WrongType)?;
1629        }
1630        ComponentInstance::GlobalComponent(glob) => {
1631            glob.as_ref().set_property(name, value)?;
1632        }
1633    }
1634    Ok(())
1635}

1637/// Return true if the Value can be used for a property of the given type
1638fn check_value_type(value: &Value, ty: &Type) -> bool {
1639    match ty {
1640        Type::Void => true,
1641        Type::Invalid
1642        | Type::InferredProperty
1643        | Type::InferredCallback
1644        | Type::Callback { .. }
1645        | Type::Function { .. }
1646        | Type::ElementReference => panic!("not valid property type"),
1647        Type::Float32 => matches!(value, Value::Number(_)),
1648        Type::Int32 => matches!(value, Value::Number(_)),
1649        Type::String => matches!(value, Value::String(_)),
1650        Type::Color => matches!(value, Value::Brush(_)),
1651        Type::UnitProduct(_)
1652        | Type::Duration
1653        | Type::PhysicalLength
1654        | Type::LogicalLength
1655        | Type::Rem
1656        | Type::Angle
1657        | Type::Percent => matches!(value, Value::Number(_)),
1658        Type::Image => matches!(value, Value::Image(_)),
1659        Type::Bool => matches!(value, Value::Bool(_)),
1660        Type::Model => {
1661            matches!(value, Value::Model(_) | Value::Bool(_) | Value::Number(_))
1662        }
1663        Type::PathData => matches!(value, Value::PathData(_)),
1664        Type::Easing => matches!(value, Value::EasingCurve(_)),
1665        Type::Brush => matches!(value, Value::Brush(_)),
1666        Type::Array(inner) => {
1667            matches!(value, Value::Model(m) if m.iter().all(|v| check_value_type(&v, inner)))
1668        }
1669        Type::Struct(s) => {
1670            matches!(value, Value::Struct(str) if str.iter().all(|(k, v)| s.fields.get(k).is_some_and(|ty| check_value_type(v, ty))))
1671        }
1672        Type::Enumeration(en) => {
1673            matches!(value, Value::EnumerationValue(name, _) if name == en.name.as_str())
1674        }
1675        Type::LayoutCache => matches!(value, Value::LayoutCache(_)),
1676        Type::ComponentFactory => matches!(value, Value::ComponentFactory(_)),
1677    }

1680pub(crate) fn invoke_callback(
1681    component_instance: &ComponentInstance,
1682    element: &ElementRc,
1683    callback_name: &SmolStr,
1684    args: &[Value],
1685) -> Option<Value> {
1686    generativity::make_guard!(guard);
1687    match enclosing_component_instance_for_element(element, component_instance, guard) {
1688        ComponentInstance::InstanceRef(enclosing_component) => {
1689            let description = enclosing_component.description;
1690            let element = element.borrow();
1691            if element.id == element.enclosing_component.upgrade().unwrap().root_element.borrow().id
1692            {
1693                if let Some(callback_offset) = description.custom_callbacks.get(callback_name) {
1694                    let callback = callback_offset.apply(&*enclosing_component.instance);
1695                    let res = callback.call(args);
1696                    return Some(if res != Value::Void {
1697                        res
1698                    } else if let Some(Type::Callback(callback)) = description
1699                        .original
1700                        .root_element
1701                        .borrow()
1702                        .property_declarations
1703                        .get(callback_name)
1704                        .map(|d| &d.property_type)
1705                    {
1706                        // If the callback was not set, the return value will be Value::Void, but we need
1707                        // to make sure that the value is actually of the right type as returned by the
1708                        // callback, otherwise we will get panics later
1709                        default_value_for_type(&callback.return_type)
1710                    } else {
1711                        res
1712                    });
1713                } else if enclosing_component.description.original.is_global() {
1714                    return None;
1715                }
1716            };
1717            let item_info = &description.items[element.id.as_str()];
1718            let item = unsafe { item_info.item_from_item_tree(enclosing_component.as_ptr()) };
1719            item_info
1720                .rtti
1721                .callbacks
1722                .get(callback_name.as_str())
1723                .map(|callback| callback.call(item, args))
1724        }
1725        ComponentInstance::GlobalComponent(global) => {
1726            Some(global.as_ref().invoke_callback(callback_name, args).unwrap())
1727        }
1728    }
1729}

1731pub(crate) fn set_callback_handler(
1732    component_instance: &ComponentInstance,
1733    element: &ElementRc,
1734    callback_name: &str,
1735    handler: CallbackHandler,
1736) -> Result<(), ()> {
1737    generativity::make_guard!(guard);
1738    match enclosing_component_instance_for_element(element, component_instance, guard) {
1739        ComponentInstance::InstanceRef(enclosing_component) => {
1740            let description = enclosing_component.description;
1741            let element = element.borrow();
1742            if element.id == element.enclosing_component.upgrade().unwrap().root_element.borrow().id
1743            {
1744                if let Some(callback_offset) = description.custom_callbacks.get(callback_name) {
1745                    let callback = callback_offset.apply(&*enclosing_component.instance);
1746                    callback.set_handler(handler);
1747                    return Ok(());
1748                } else if enclosing_component.description.original.is_global() {
1749                    return Err(());
1750                }
1751            };
1752            let item_info = &description.items[element.id.as_str()];
1753            let item = unsafe { item_info.item_from_item_tree(enclosing_component.as_ptr()) };
1754            if let Some(callback) = item_info.rtti.callbacks.get(callback_name) {
1755                callback.set_handler(item, handler);
1756                Ok(())
1757            } else {
1758                Err(())
1759            }
1760        }
1761        ComponentInstance::GlobalComponent(global) => {
1762            global.as_ref().set_callback_handler(callback_name, handler)
1763        }
1764    }
1765}

1790/// Return the component instance which hold the given element.
1791/// Does not take in account the global component.
1792pub fn enclosing_component_for_element<'a, 'old_id, 'new_id>(
1793    element: &'a ElementRc,
1794    component: InstanceRef<'a, 'old_id>,
1795    _guard: generativity::Guard<'new_id>,
1796) -> InstanceRef<'a, 'new_id> {
1797    let enclosing = &element.borrow().enclosing_component.upgrade().unwrap();
1798    if Rc::ptr_eq(enclosing, &component.description.original) {
1799        // Safety: new_id is an unique id
1800        unsafe {
1801            std::mem::transmute::<InstanceRef<'a, 'old_id>, InstanceRef<'a, 'new_id>>(component)
1802        }
1803    } else {
1804        assert!(!enclosing.is_global());
1805        // Safety: this is the only place we use this 'static lifetime in this function and nothing is returned with it
1806        // For some reason we can't make a new guard here because the compiler thinks we are returning that
1807        // (it assumes that the 'id must outlive 'a , which is not true)
1808        let static_guard = unsafe { generativity::Guard::new(generativity::Id::<'static>::new()) };
1809
1810        let parent_instance = component.parent_instance(static_guard).unwrap();
1811        enclosing_component_for_element(element, parent_instance, _guard)
1812    }
1813}

1815/// Return the component instance which hold the given element.
1816/// The difference with enclosing_component_for_element is that it takes the GlobalComponent into account.
1817pub(crate) fn enclosing_component_instance_for_element<'a, 'new_id>(
1818    element: &'a ElementRc,
1819    component_instance: &ComponentInstance<'a, '_>,
1820    guard: generativity::Guard<'new_id>,
1821) -> ComponentInstance<'a, 'new_id> {
1822    let enclosing = &element.borrow().enclosing_component.upgrade().unwrap();
1823    match component_instance {
1824        ComponentInstance::InstanceRef(component) => {
1825            if enclosing.is_global() && !Rc::ptr_eq(enclosing, &component.description.original) {
1826                let root = component.toplevel_instance(guard);
1827                ComponentInstance::GlobalComponent(
1828                    root.description
1829                        .extra_data_offset
1830                        .apply(root.instance.get_ref())
1831                        .globals
1832                        .get()
1833                        .unwrap()
1834                        .get(enclosing.root_element.borrow().id.as_str())
1835                        .unwrap(),
1836                )
1837            } else {
1838                ComponentInstance::InstanceRef(enclosing_component_for_element(
1839                    element, *component, guard,
1840                ))
1841            }
1842        }
1843        ComponentInstance::GlobalComponent(global) => {
1844            //assert!(Rc::ptr_eq(enclosing, &global.component));
1845            ComponentInstance::GlobalComponent(global.clone())
1846        }
1847    }
1848}

1850pub fn new_struct_with_bindings<ElementType: 'static + Default + corelib::rtti::BuiltinItem>(
1851    bindings: &i_slint_compiler::object_tree::BindingsMap,
1852    local_context: &mut EvalLocalContext,
1853) -> ElementType {
1854    let mut element = ElementType::default();
1855    for (prop, info) in ElementType::fields::<Value>().into_iter() {
1856        if let Some(binding) = &bindings.get(prop) {
1857            let value = eval_expression(&binding.borrow(), local_context);
1858            info.set_field(&mut element, value).unwrap();
1859        }
1860    }
1861    element
1862}

1864fn convert_from_lyon_path<'a>(
1865    events_it: impl IntoIterator<Item = &'a i_slint_compiler::expression_tree::Expression>,
1866    points_it: impl IntoIterator<Item = &'a i_slint_compiler::expression_tree::Expression>,
1867    local_context: &mut EvalLocalContext,
1868) -> PathData {
1869    let events = events_it
1870        .into_iter()
1871        .map(|event_expr| eval_expression(event_expr, local_context).try_into().unwrap())
1872        .collect::<SharedVector<_>>();
1873
1874    let points = points_it
1875        .into_iter()
1876        .map(|point_expr| {
1877            let point_value = eval_expression(point_expr, local_context);
1878            let point_struct: Struct = point_value.try_into().unwrap();
1879            let mut point = i_slint_core::graphics::Point::default();
1880            let x: f64 = point_struct.get_field("x").unwrap().clone().try_into().unwrap();
1881            let y: f64 = point_struct.get_field("y").unwrap().clone().try_into().unwrap();
1882            point.x = x as _;
1883            point.y = y as _;
1884            point
1885        })
1886        .collect::<SharedVector<_>>();
1887
1888    PathData::Events(events, points)
1889}

1891pub fn convert_path(path: &ExprPath, local_context: &mut EvalLocalContext) -> PathData {
1892    match path {
1893        ExprPath::Elements(elements) => PathData::Elements(
1894            elements
1895                .iter()
1896                .map(|element| convert_path_element(element, local_context))
1897                .collect::<SharedVector<PathElement>>(),
1898        ),
1899        ExprPath::Events(events, points) => {
1900            convert_from_lyon_path(events.iter(), points.iter(), local_context)
1901        }
1902        ExprPath::Commands(commands) => {
1903            if let Value::String(commands) = eval_expression(commands, local_context) {
1904                PathData::Commands(commands)
1905            } else {
1906                panic!("binding to path commands does not evaluate to string");
1907            }
1908        }
1909    }
1910}

1912fn convert_path_element(
1913    expr_element: &ExprPathElement,
1914    local_context: &mut EvalLocalContext,
1915) -> PathElement {
1916    match expr_element.element_type.native_class.class_name.as_str() {
1917        "MoveTo" => {
1918            PathElement::MoveTo(new_struct_with_bindings(&expr_element.bindings, local_context))
1919        }
1920        "LineTo" => {
1921            PathElement::LineTo(new_struct_with_bindings(&expr_element.bindings, local_context))
1922        }
1923        "ArcTo" => {
1924            PathElement::ArcTo(new_struct_with_bindings(&expr_element.bindings, local_context))
1925        }
1926        "CubicTo" => {
1927            PathElement::CubicTo(new_struct_with_bindings(&expr_element.bindings, local_context))
1928        }
1929        "QuadraticTo" => PathElement::QuadraticTo(new_struct_with_bindings(
1930            &expr_element.bindings,
1931            local_context,
1932        )),
1933        "Close" => PathElement::Close,
1934        _ => panic!(
1935            "Cannot create unsupported path element {}",
1936            expr_element.element_type.native_class.class_name
1937        ),
1938    }

1941/// Create a value suitable as the default value of a given type
1942pub fn default_value_for_type(ty: &Type) -> Value {
1943    match ty {
1944        Type::Float32 | Type::Int32 => Value::Number(0.),
1945        Type::String => Value::String(Default::default()),
1946        Type::Color | Type::Brush => Value::Brush(Default::default()),
1947        Type::Duration | Type::Angle | Type::PhysicalLength | Type::LogicalLength | Type::Rem => {
1948            Value::Number(0.)
1949        }
1950        Type::Image => Value::Image(Default::default()),
1951        Type::Bool => Value::Bool(false),
1952        Type::Callback { .. } => Value::Void,
1953        Type::Struct(s) => Value::Struct(
1954            s.fields
1955                .iter()
1956                .map(|(n, t)| (n.to_string(), default_value_for_type(t)))
1957                .collect::<Struct>(),
1958        ),
1959        Type::Array(_) | Type::Model => Value::Model(Default::default()),
1960        Type::Percent => Value::Number(0.),
1961        Type::Enumeration(e) => Value::EnumerationValue(
1962            e.name.to_string(),
1963            e.values.get(e.default_value).unwrap().to_string(),
1964        ),
1965        Type::Easing => Value::EasingCurve(Default::default()),
1966        Type::Void | Type::Invalid => Value::Void,
1967        Type::UnitProduct(_) => Value::Number(0.),
1968        Type::PathData => Value::PathData(Default::default()),
1969        Type::LayoutCache => Value::LayoutCache(Default::default()),
1970        Type::ComponentFactory => Value::ComponentFactory(Default::default()),
1971        Type::InferredProperty
1972        | Type::InferredCallback
1973        | Type::ElementReference
1974        | Type::Function { .. } => {
1975            panic!("There can't be such property")
1976        }
1977    }
1978}

1980pub struct MenuWrapper {
1981    entries: Expression,
1982    sub_menu: NamedReference,
1983    activated: NamedReference,
1984    item_tree: crate::dynamic_item_tree::ErasedItemTreeBoxWeak,
1985}
1986i_slint_core::MenuVTable_static!(static MENU_WRAPPER_VTABLE for MenuWrapper);
1987impl Menu for MenuWrapper {
1988    fn sub_menu(&self, parent: Option<&MenuEntry>, result: &mut SharedVector<MenuEntry>) {
1989        let Some(s) = self.item_tree.upgrade() else { return };
1990        generativity::make_guard!(guard);
1991        let compo_box = s.unerase(guard);
1992        let instance_ref = compo_box.borrow_instance();
1993        let res = match parent {
1994            None => eval_expression(
1995                &self.entries,
1996                &mut EvalLocalContext::from_component_instance(instance_ref),
1997            ),
1998            Some(parent) => {
1999                let instance_ref = ComponentInstance::InstanceRef(instance_ref);
2000                invoke_callback(
2001                    &instance_ref,
2002                    &self.sub_menu.element(),
2003                    self.sub_menu.name(),
2004                    &[parent.clone().into()],
2005                )
2006                .unwrap()
2007            }
2008        };
2009        let Value::Model(model) = res else { panic!("Not a model of menu entries {res:?}") };
2010        *result = model.iter().map(|v| v.try_into().unwrap()).collect();
2011    }
2012    fn activate(&self, entry: &MenuEntry) {
2013        let Some(s) = self.item_tree.upgrade() else { return };
2014        generativity::make_guard!(guard);
2015        let compo_box = s.unerase(guard);
2016        let instance_ref = compo_box.borrow_instance();
2017        let instance_ref = ComponentInstance::InstanceRef(instance_ref);
2018        invoke_callback(
2019            &instance_ref,
2020            &self.activated.element(),
2021            self.activated.name(),
2022            &[entry.clone().into()],
2023        )
2024        .unwrap();
2025    }
2026}
