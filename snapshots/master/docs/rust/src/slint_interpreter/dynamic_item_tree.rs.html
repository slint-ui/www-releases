128impl RepeatedItemTree for ErasedItemTreeBox {
129    type Data = Value;
130
131    fn update(&self, index: usize, data: Self::Data) {
132        generativity::make_guard!(guard);
133        let s = self.unerase(guard);
134        let is_repeated = s.description.original.parent_element.upgrade().is_some_and(|p| {
135            p.borrow().repeated.as_ref().is_some_and(|r| !r.is_conditional_element)
136        });
137        if is_repeated {
138            s.description.set_property(s.borrow(), SPECIAL_PROPERTY_INDEX, index.into()).unwrap();
139            s.description.set_property(s.borrow(), SPECIAL_PROPERTY_MODEL_DATA, data).unwrap();
140        }
141    }
142
143    fn init(&self) {
144        self.run_setup_code();
145    }
146
147    fn listview_layout(self: Pin<&Self>, offset_y: &mut LogicalLength) -> LogicalLength {
148        generativity::make_guard!(guard);
149        let s = self.unerase(guard);
150
151        let geom = s.description.original.root_element.borrow().geometry_props.clone().unwrap();
152
153        crate::eval::store_property(
154            s.borrow_instance(),
155            &geom.y.element(),
156            geom.y.name(),
157            Value::Number(offset_y.get() as f64),
158        )
159        .expect("cannot set y");
160
161        let h: LogicalLength = crate::eval::load_property(
162            s.borrow_instance(),
163            &geom.height.element(),
164            geom.height.name(),
165        )
166        .expect("missing height")
167        .try_into()
168        .expect("height not the right type");
169
170        *offset_y += h;
171        LogicalLength::new(self.borrow().as_ref().layout_info(Orientation::Horizontal).min)
172    }
173
174    fn layout_item_info(self: Pin<&Self>, o: Orientation) -> LayoutItemInfo {
175        LayoutItemInfo { constraint: self.borrow().as_ref().layout_info(o) }
176    }

440fn internal_properties_to_public<'a>(
441    prop_iter: impl Iterator<Item = (&'a SmolStr, &'a PropertyDeclaration)> + 'a,
442) -> impl Iterator<
443    Item = (
444        SmolStr,
445        i_slint_compiler::langtype::Type,
446        i_slint_compiler::object_tree::PropertyVisibility,
447    ),
448> + 'a {
449    prop_iter.filter(|(_, v)| v.expose_in_public_api).map(|(s, v)| {
450        let name = v
451            .node
452            .as_ref()
453            .and_then(|n| {
454                n.child_node(parser::SyntaxKind::DeclaredIdentifier)
455                    .and_then(|n| n.child_token(parser::SyntaxKind::Identifier))
456            })
457            .map(|n| n.to_smolstr())
458            .unwrap_or_else(|| s.to_smolstr());
459        (name, v.property_type.clone(), v.visibility)
460    })
461}

544    /// Set a value to property.
545    ///
546    /// Return an error if the property with this name does not exist,
547    /// or if the value is the wrong type.
548    /// Panics if the component is not an instance corresponding to this ItemTreeDescription,
549    pub fn set_property(
550        &self,
551        component: ItemTreeRefPin,
552        name: &str,
553        value: Value,
554    ) -> Result<(), crate::api::SetPropertyError> {
555        if !core::ptr::eq((&self.ct) as *const _, component.get_vtable() as *const _) {
556            panic!("mismatch instance and vtable");
557        }
558        generativity::make_guard!(guard);
559        let c = unsafe { InstanceRef::from_pin_ref(component, guard) };
560        if let Some(alias) = self
561            .original
562            .root_element
563            .borrow()
564            .property_declarations
565            .get(name)
566            .and_then(|d| d.is_alias.as_ref())
567        {
568            eval::store_property(c, &alias.element(), alias.name(), value)
569        } else {
570            eval::store_property(c, &self.original.root_element, name, value)
571        }
572    }

574    /// Set a binding to a property
575    ///
576    /// Returns an error if the instance does not corresponds to this ItemTreeDescription,
577    /// or if the property with this name does not exist in this component
578    pub fn set_binding(
579        &self,
580        component: ItemTreeRefPin,
581        name: &str,
582        binding: Box<dyn Fn() -> Value>,
583    ) -> Result<(), ()> {
584        if !core::ptr::eq((&self.ct) as *const _, component.get_vtable() as *const _) {
585            return Err(());
586        }
587        let x = self.custom_properties.get(name).ok_or(())?;
588        unsafe {
589            x.prop
590                .set_binding(
591                    Pin::new_unchecked(&*component.as_ptr().add(x.offset)),
592                    binding,
593                    i_slint_core::rtti::AnimatedBindingKind::NotAnimated,
594                )
595                .unwrap()
596        };
597        Ok(())
598    }

605        if !core::ptr::eq((&self.ct) as *const _, component.get_vtable() as *const _) {
606            return Err(());

609        // Safety: we just verified that the component has the right vtable

612            .original
613            .root_element
614            .borrow()
615            .property_declarations

617            .and_then(|d| d.is_alias.as_ref())

659    /// Invoke the specified callback or function
660    ///
661    /// Returns an error if the component is not an instance corresponding to this ItemTreeDescription,
662    /// or if the callback with this name does not exist in this component
663    pub fn invoke(
664        &self,
665        component: ItemTreeRefPin,
666        name: &SmolStr,
667        args: &[Value],
668    ) -> Result<Value, ()> {
669        if !core::ptr::eq((&self.ct) as *const _, component.get_vtable() as *const _) {
670            return Err(());
671        }
672        generativity::make_guard!(guard);
673        // Safety: we just verified that the component has the right vtable
674        let c = unsafe { InstanceRef::from_pin_ref(component, guard) };
675        let borrow = self.original.root_element.borrow();
676        let decl = borrow.property_declarations.get(name).ok_or(())?;
677
678        let (elem, name) = if let Some(alias) = &decl.is_alias {
679            (alias.element(), alias.name())
680        } else {
681            (self.original.root_element.clone(), name)
682        };
683
684        let inst = eval::ComponentInstance::InstanceRef(c);
685
686        if matches!(&decl.property_type, Type::Function { .. }) {
687            eval::call_function(&inst, &elem, name, args.to_vec()).ok_or(())
688        } else {
689            eval::invoke_callback(&inst, &elem, name, args).ok_or(())
690        }
691    }

723#[cfg_attr(not(feature = "ffi"), i_slint_core_macros::remove_extern)]
724extern "C" fn visit_children_item(
725    component: ItemTreeRefPin,
726    index: isize,
727    order: TraversalOrder,
728    v: ItemVisitorRefMut,
729) -> VisitChildrenResult {
730    generativity::make_guard!(guard);
731    let instance_ref = unsafe { InstanceRef::from_pin_ref(component, guard) };
732    let comp_rc = instance_ref.self_weak().get().unwrap().upgrade().unwrap();
733    i_slint_core::item_tree::visit_item_tree(
734        instance_ref.instance,
735        &vtable::VRc::into_dyn(comp_rc),
736        get_item_tree(component).as_slice(),
737        index,
738        order,
739        v,
740        |_, order, visitor, index| {
741            if index as usize >= instance_ref.description.repeater.len() {
742                // Do nothing: We are ComponentContainer and Our parent already did all the work!
743                VisitChildrenResult::CONTINUE
744            } else {
745                // `ensure_updated` needs a 'static lifetime so we must call get_untagged.
746                // Safety: we do not mix the component with other component id in this function
747                let rep_in_comp =
748                    unsafe { instance_ref.description.repeater[index as usize].get_untagged() };
749                ensure_repeater_updated(instance_ref, rep_in_comp);
750                let repeater = rep_in_comp.offset.apply_pin(instance_ref.instance);
751                repeater.visit(order, visitor)
752            }
753        },
754    )
755}

757/// Make sure that the repeater is updated
758fn ensure_repeater_updated<'id>(
759    instance_ref: InstanceRef<'_, 'id>,
760    rep_in_comp: &RepeaterWithinItemTree<'id, '_>,
761) {
762    let repeater = rep_in_comp.offset.apply_pin(instance_ref.instance);
763    let init = || {
764        instantiate(
765            rep_in_comp.item_tree_to_repeat.clone(),
766            instance_ref.self_weak().get().cloned(),
767            None,
768            None,
769            Default::default(),
770        )
771    };
772    if let Some(lv) = &rep_in_comp
773        .item_tree_to_repeat
774        .original
775        .parent_element
776        .upgrade()
777        .unwrap()
778        .borrow()
779        .repeated
780        .as_ref()
781        .unwrap()
782        .is_listview
783    {
784        let assume_property_logical_length =
785            |prop| unsafe { Pin::new_unchecked(&*(prop as *const Property<LogicalLength>)) };
786        let get_prop = |nr: &NamedReference| -> LogicalLength {
787            eval::load_property(instance_ref, &nr.element(), nr.name()).unwrap().try_into().unwrap()
788        };
789        repeater.ensure_updated_listview(
790            init,
791            assume_property_logical_length(get_property_ptr(&lv.viewport_width, instance_ref)),
792            assume_property_logical_length(get_property_ptr(&lv.viewport_height, instance_ref)),
793            assume_property_logical_length(get_property_ptr(&lv.viewport_y, instance_ref)),
794            get_prop(&lv.listview_width),
795            assume_property_logical_length(get_property_ptr(&lv.listview_height, instance_ref)),
796        );
797    } else {
798        repeater.ensure_updated(init);
799    }
800}

827/// Create a ItemTreeDescription from a source.
828/// The path corresponding to the source need to be passed as well (path is used for diagnostics
829/// and loading relative assets)
830pub async fn load(
831    source: String,
832    path: std::path::PathBuf,
833    mut compiler_config: CompilerConfiguration,
834) -> CompilationResult {
835    // If the native style should be Qt, resolve it here as we know that we have it
836    let is_native = match &compiler_config.style {
837        Some(s) => s == "native",
838        None => std::env::var("SLINT_STYLE").map_or(true, |s| s == "native"),
839    };
840    if is_native {
841        // On wasm, look at the browser user agent
842        #[cfg(target_arch = "wasm32")]
843        let target = web_sys::window()
844            .and_then(|window| window.navigator().platform().ok())
845            .map_or("wasm", |platform| {
846                let platform = platform.to_ascii_lowercase();
847                if platform.contains("mac")
848                    || platform.contains("iphone")
849                    || platform.contains("ipad")
850                {
851                    "apple"
852                } else if platform.contains("android") {
853                    "android"
854                } else if platform.contains("win") {
855                    "windows"
856                } else if platform.contains("linux") {
857                    "linux"
858                } else {
859                    "wasm"
860                }
861            });
862        #[cfg(not(target_arch = "wasm32"))]
863        let target = "";
864        compiler_config.style = Some(
865            i_slint_common::get_native_style(i_slint_backend_selector::HAS_NATIVE_STYLE, target)
866                .to_string(),
867        );
868    }
869
870    let diag = BuildDiagnostics::default();
871    #[cfg(feature = "internal-highlight")]
872    let (path, mut diag, loader, raw_type_loader) =
873        i_slint_compiler::load_root_file_with_raw_type_loader(
874            &path,
875            &path,
876            source,
877            diag,
878            compiler_config,
879        )
880        .await;
881    #[cfg(not(feature = "internal-highlight"))]
882    let (path, mut diag, loader) =
883        i_slint_compiler::load_root_file(&path, &path, source, diag, compiler_config).await;
884    if diag.has_errors() {
885        return CompilationResult {
886            components: HashMap::new(),
887            diagnostics: diag.into_iter().collect(),
888            #[cfg(feature = "internal")]
889            structs_and_enums: Vec::new(),
890            #[cfg(feature = "internal")]
891            named_exports: Vec::new(),
892        };
893    }
894
895    #[cfg(feature = "internal-highlight")]
896    let loader = Rc::new(loader);
897    #[cfg(feature = "internal-highlight")]
898    let raw_type_loader = raw_type_loader.map(Rc::new);
899
900    let doc = loader.get_document(&path).unwrap();
901
902    let compiled_globals = Rc::new(CompiledGlobalCollection::compile(doc));
903    let mut components = HashMap::new();
904
905    let popup_menu_description = if let Some(popup_menu_impl) = &doc.popup_menu_impl {
906        PopupMenuDescription::Rc(Rc::new_cyclic(|weak| {
907            generativity::make_guard!(guard);
908            ErasedItemTreeDescription::from(generate_item_tree(
909                popup_menu_impl,
910                Some(compiled_globals.clone()),
911                PopupMenuDescription::Weak(weak.clone()),
912                true,
913                guard,
914            ))
915        }))
916    } else {
917        PopupMenuDescription::Weak(Default::default())
918    };
919
920    for c in doc.exported_roots() {
921        generativity::make_guard!(guard);
922        #[allow(unused_mut)]
923        let mut it = generate_item_tree(
924            &c,
925            Some(compiled_globals.clone()),
926            popup_menu_description.clone(),
927            false,
928            guard,
929        );
930        #[cfg(feature = "internal-highlight")]
931        {
932            let _ = it.type_loader.set(loader.clone());
933            let _ = it.raw_type_loader.set(raw_type_loader.clone());
934        }
935        components.insert(c.id.to_string(), ComponentDefinition { inner: it.into() });
936    }
937
938    if components.is_empty() {
939        diag.push_error_with_span("No component found".into(), Default::default());
940    };
941
942    #[cfg(feature = "internal")]
943    let structs_and_enums = doc.used_types.borrow().structs_and_enums.clone();
944
945    #[cfg(feature = "internal")]
946    let named_exports = doc
947        .exports
948        .iter()
949        .filter_map(|export| match &export.1 {
950            Either::Left(component) if !component.is_global() => {
951                Some((&export.0.name, &component.id))
952            }
953            Either::Right(ty) => match &ty {
954                Type::Struct(s) if s.node().is_some() => {
955                    if let StructName::User { name, .. } = &s.name {
956                        Some((&export.0.name, name))
957                    } else {
958                        None
959                    }
960                }
961                Type::Enumeration(en) => Some((&export.0.name, &en.name)),
962                _ => None,
963            },
964            _ => None,
965        })
966        .filter(|(export_name, type_name)| *export_name != *type_name)
967        .map(|(export_name, type_name)| (type_name.to_string(), export_name.to_string()))
968        .collect::<Vec<_>>();
969
970    CompilationResult {
971        diagnostics: diag.into_iter().collect(),
972        components,
973        #[cfg(feature = "internal")]
974        structs_and_enums,
975        #[cfg(feature = "internal")]
976        named_exports,
977    }
978}

1207            Box::new(unsafe {
1208                rtti::MaybeAnimatedPropertyInfoWrapper(
1209                    vtable::FieldOffset::<u8, Property<T>, _>::new_from_offset_pinned(0),
1210                )
1211            }),
1212            dynamic_type::StaticTypeInfo::new::<Property<T>>(),
1213        )
1214    }
1215
1216    fn property_info_for_type(
1217        ty: &Type,
1218        name: &str,
1219    ) -> Option<(Box<dyn PropertyInfo<u8, Value>>, dynamic_type::StaticTypeInfo)> {
1220        Some(match ty {
1221            Type::Float32 => animated_property_info::<f32>(),
1222            Type::Int32 => animated_property_info::<i32>(),
1223            Type::String => property_info::<SharedString>(),
1224            Type::Color => animated_property_info::<Color>(),
1225            Type::Brush => animated_property_info::<Brush>(),
1226            Type::Duration => animated_property_info::<i64>(),
1227            Type::Angle => animated_property_info::<f32>(),
1228            Type::PhysicalLength => animated_property_info::<f32>(),
1229            Type::LogicalLength => animated_property_info::<f32>(),
1230            Type::Rem => animated_property_info::<f32>(),
1231            Type::Image => property_info::<i_slint_core::graphics::Image>(),
1232            Type::Bool => property_info::<bool>(),
1233            Type::ComponentFactory => property_info::<ComponentFactory>(),
1234            Type::Struct(s)
1235                if matches!(
1236                    s.name,
1237                    StructName::BuiltinPrivate(BuiltinPrivateStruct::StateInfo)
1238                ) =>
1239            {
1240                property_info::<i_slint_core::properties::StateInfo>()
1241            }
1242            Type::Struct(_) => property_info::<Value>(),
1243            Type::Array(_) => property_info::<Value>(),
1244            Type::Easing => property_info::<i_slint_core::animations::EasingCurve>(),
1245            Type::Percent => animated_property_info::<f32>(),
1246            Type::Enumeration(e) => {
1247                macro_rules! match_enum_type {
1248                    ($( $(#[$enum_doc:meta])* enum $Name:ident { $($body:tt)* })*) => {
1249                        match e.name.as_str() {
1250                            $(
1251                                stringify!($Name) => property_info::<i_slint_core::items::$Name>(),
1252                            )*
1253                            x => unreachable!("Unknown non-builtin enum {x}"),
1254                        }
1255                    }
1256                }
1257                if e.node.is_some() {
1258                    property_info::<Value>()
1259                } else {
1260                    i_slint_common::for_each_enums!(match_enum_type)
1261                }
1262            }
1263            Type::LayoutCache => property_info::<SharedVector<f32>>(),
1264            Type::ArrayOfU16 => property_info::<SharedVector<u16>>(),
1265            Type::Function { .. } | Type::Callback { .. } => return None,
1266            Type::StyledText => property_info::<StyledText>(),
1267            // These can't be used in properties
1268            Type::Invalid
1269            | Type::Void
1270            | Type::InferredProperty
1271            | Type::InferredCallback
1272            | Type::Model
1273            | Type::PathData
1274            | Type::UnitProduct(_)
1275            | Type::ElementReference => panic!("bad type {ty:?} for property {name}"),
1276        })
1277    }
1278
1279    for (name, decl) in &component.root_element.borrow().property_declarations {
1280        if decl.is_alias.is_some() {
1281            continue;
1282        }
1283        if matches!(&decl.property_type, Type::Callback { .. }) {
1284            custom_callbacks
1285                .insert(name.clone(), builder.type_builder.add_field_type::<Callback>());
1286            continue;
1287        }
1288        let Some((prop, type_info)) = property_info_for_type(&decl.property_type, name) else {
1289            continue;
1290        };
1291        custom_properties.insert(
1292            name.clone(),
1293            PropertiesWithinComponent { offset: builder.type_builder.add_field(type_info), prop },
1294        );
1295    }
1296    if let Some(parent_element) = component.parent_element.upgrade()
1297        && let Some(r) = &parent_element.borrow().repeated
1298        && !r.is_conditional_element
1299    {
1300        let (prop, type_info) = property_info::<u32>();
1301        custom_properties.insert(
1302            SPECIAL_PROPERTY_INDEX.into(),
1303            PropertiesWithinComponent { offset: builder.type_builder.add_field(type_info), prop },
1304        );
1305
1306        let model_ty =
1307            Expression::RepeaterModelReference { element: component.parent_element.clone() }.ty();
1308        let (prop, type_info) =
1309            property_info_for_type(&model_ty, SPECIAL_PROPERTY_MODEL_DATA).unwrap();
1310        custom_properties.insert(
1311            SPECIAL_PROPERTY_MODEL_DATA.into(),
1312            PropertiesWithinComponent { offset: builder.type_builder.add_field(type_info), prop },
1313        );
1314    }
1315
1316    let parent_item_tree_offset =
1317        if component.parent_element.upgrade().is_some() || is_popup_menu_impl {
1318            Some(builder.type_builder.add_field_type::<OnceCell<ErasedItemTreeBoxWeak>>())
1319        } else {
1320            None
1321        };
1322
1323    let root_offset = builder.type_builder.add_field_type::<OnceCell<ErasedItemTreeBoxWeak>>();
1324
1325    let window_adapter_offset = builder.type_builder.add_field_type::<OnceCell<WindowAdapterRc>>();
1326
1327    let extra_data_offset = builder.type_builder.add_field_type::<ComponentExtraData>();
1328
1329    let change_trackers = (!builder.change_callbacks.is_empty()).then(|| {
1330        (
1331            builder.type_builder.add_field_type::<OnceCell<Vec<ChangeTracker>>>(),
1332            builder.change_callbacks,
1333        )
1334    });
1335    let timers = component
1336        .timers
1337        .borrow()
1338        .iter()
1339        .map(|_| builder.type_builder.add_field_type::<Timer>())
1340        .collect();
1341
1342    // only the public exported component needs the public property list
1343    let public_properties = if component.parent_element.upgrade().is_none() {
1344        component.root_element.borrow().property_declarations.clone()
1345    } else {
1346        Default::default()
1347    };
1348
1349    let t = ItemTreeVTable {
1350        visit_children_item,
1351        layout_info,
1352        get_item_ref,
1353        get_item_tree,
1354        get_subtree_range,
1355        get_subtree,
1356        parent_node,
1357        embed_component,
1358        subtree_index,
1359        item_geometry,
1360        accessible_role,
1361        accessible_string_property,
1362        accessibility_action,
1363        supported_accessibility_actions,
1364        item_element_infos,
1365        window_adapter,
1366        drop_in_place,
1367        dealloc,
1368    };
1369    let t = ItemTreeDescription {
1370        ct: t,
1371        dynamic_type: builder.type_builder.build(),
1372        item_tree: builder.tree_array,
1373        item_array: builder.item_array,
1374        items: builder.items_types,
1375        custom_properties,
1376        custom_callbacks,
1377        original: component.clone(),
1378        original_elements: builder.original_elements,
1379        repeater: builder.repeater,
1380        repeater_names: builder.repeater_names,
1381        parent_item_tree_offset,
1382        root_offset,
1383        window_adapter_offset,
1384        extra_data_offset,
1385        public_properties,
1386        compiled_globals,
1387        change_trackers,
1388        timers,
1389        popup_ids: std::cell::RefCell::new(HashMap::new()),
1390        popup_menu_description: builder.popup_menu_description,
1391        #[cfg(feature = "internal-highlight")]
1392        type_loader: std::cell::OnceCell::new(),
1393        #[cfg(feature = "internal-highlight")]
1394        raw_type_loader: std::cell::OnceCell::new(),
1395        debug_handler: std::cell::RefCell::new(Rc::new(|_, text| {
1396            i_slint_core::debug_log!("{text}")
1397        })),
1398    };
1399
1400    Rc::new(t)
1401}

1403pub fn animation_for_property(
1404    component: InstanceRef,
1405    animation: &Option<i_slint_compiler::object_tree::PropertyAnimation>,
1406) -> AnimatedBindingKind {
1407    match animation {
1408        Some(i_slint_compiler::object_tree::PropertyAnimation::Static(anim_elem)) => {
1409            AnimatedBindingKind::Animation(eval::new_struct_with_bindings(
1410                &anim_elem.borrow().bindings,
1411                &mut eval::EvalLocalContext::from_component_instance(component),
1412            ))
1413        }
1414        Some(i_slint_compiler::object_tree::PropertyAnimation::Transition {
1415            animations,
1416            state_ref,
1417        }) => {
1418            let component_ptr = component.as_ptr();
1419            let vtable = NonNull::from(&component.description.ct).cast();
1420            let animations = animations.clone();
1421            let state_ref = state_ref.clone();
1422            AnimatedBindingKind::Transition(Box::new(
1423                move || -> (PropertyAnimation, i_slint_core::animations::Instant) {
1424                    generativity::make_guard!(guard);
1425                    let component = unsafe {
1426                        InstanceRef::from_pin_ref(
1427                            Pin::new_unchecked(vtable::VRef::from_raw(
1428                                vtable,
1429                                NonNull::new_unchecked(component_ptr as *mut u8),
1430                            )),
1431                            guard,
1432                        )
1433                    };
1434
1435                    let mut context = eval::EvalLocalContext::from_component_instance(component);
1436                    let state = eval::eval_expression(&state_ref, &mut context);
1437                    let state_info: i_slint_core::properties::StateInfo = state.try_into().unwrap();
1438                    for a in &animations {
1439                        let is_previous_state = a.state_id == state_info.previous_state;
1440                        let is_current_state = a.state_id == state_info.current_state;
1441                        match (a.direction, is_previous_state, is_current_state) {
1442                            (TransitionDirection::In, false, true)
1443                            | (TransitionDirection::Out, true, false)
1444                            | (TransitionDirection::InOut, false, true)
1445                            | (TransitionDirection::InOut, true, false) => {
1446                                return (
1447                                    eval::new_struct_with_bindings(
1448                                        &a.animation.borrow().bindings,
1449                                        &mut context,
1450                                    ),
1451                                    state_info.change_time,
1452                                );
1453                            }
1454                            _ => {}
1455                        }
1456                    }
1457                    Default::default()
1458                },
1459            ))
1460        }
1461        None => AnimatedBindingKind::NotAnimated,
1462    }
1463}

1465fn make_callback_eval_closure(
1466    expr: Expression,
1467    self_weak: ErasedItemTreeBoxWeak,
1468) -> impl Fn(&[Value]) -> Value {
1469    move |args| {
1470        let self_rc = self_weak.upgrade().unwrap();
1471        generativity::make_guard!(guard);
1472        let self_ = self_rc.unerase(guard);
1473        let instance_ref = self_.borrow_instance();
1474        let mut local_context =
1475            eval::EvalLocalContext::from_function_arguments(instance_ref, args.to_vec());
1476        eval::eval_expression(&expr, &mut local_context)
1477    }
1478}

1490            &expr,
1491            &mut eval::EvalLocalContext::from_component_instance(instance_ref),
1492        )
1493    }

1496pub fn instantiate(
1497    description: Rc<ItemTreeDescription>,
1498    parent_ctx: Option<ErasedItemTreeBoxWeak>,
1499    root: Option<ErasedItemTreeBoxWeak>,
1500    window_options: Option<&WindowOptions>,
1501    mut globals: crate::global_component::GlobalStorage,
1502) -> DynamicComponentVRc {
1503    let instance = description.dynamic_type.clone().create_instance();
1504
1505    let component_box = ItemTreeBox { instance, description: description.clone() };
1506
1507    let self_rc = vtable::VRc::new(ErasedItemTreeBox::from(component_box));
1508    let self_weak = vtable::VRc::downgrade(&self_rc);
1509
1510    generativity::make_guard!(guard);
1511    let comp = self_rc.unerase(guard);
1512    let instance_ref = comp.borrow_instance();
1513    instance_ref.self_weak().set(self_weak.clone()).ok();
1514    let description = comp.description();
1515
1516    if let Some(parent) = parent_ctx {
1517        description
1518            .parent_item_tree_offset
1519            .unwrap()
1520            .apply(instance_ref.as_ref())
1521            .set(parent)
1522            .ok()
1523            .unwrap();
1524    } else {
1525        if let Some(g) = description.compiled_globals.as_ref() {
1526            for g in g.compiled_globals.iter() {
1527                crate::global_component::instantiate(g, &mut globals, self_weak.clone());
1528            }
1529        }
1530        let extra_data = description.extra_data_offset.apply(instance_ref.as_ref());
1531        extra_data.globals.set(globals).ok().unwrap();
1532    }
1533
1534    if let Some(WindowOptions::Embed { parent_item_tree, parent_item_tree_index }) = window_options
1535    {
1536        vtable::VRc::borrow_pin(&self_rc)
1537            .as_ref()
1538            .embed_component(parent_item_tree, *parent_item_tree_index);
1539        description.root_offset.apply(instance_ref.as_ref()).set(self_weak.clone()).ok().unwrap();
1540    } else {
1541        generativity::make_guard!(guard);
1542        let root = root
1543            .or_else(|| {
1544                instance_ref.parent_instance(guard).map(|parent| parent.root_weak().clone())
1545            })
1546            .unwrap_or_else(|| self_weak.clone());
1547        description.root_offset.apply(instance_ref.as_ref()).set(root).ok().unwrap();
1548    }
1549
1550    if !description.original.is_global() {
1551        let maybe_window_adapter =
1552            if let Some(WindowOptions::UseExistingWindow(adapter)) = window_options.as_ref() {
1553                Some(adapter.clone())
1554            } else {
1555                instance_ref.maybe_window_adapter()
1556            };
1557
1558        let component_rc = vtable::VRc::into_dyn(self_rc.clone());
1559        i_slint_core::item_tree::register_item_tree(&component_rc, maybe_window_adapter);
1560    }
1561
1562    if let Some(WindowOptions::UseExistingWindow(existing_adapter)) = &window_options {
1563        description
1564            .window_adapter_offset
1565            .apply(instance_ref.as_ref())
1566            .set(existing_adapter.clone())
1567            .ok()
1568            .unwrap();
1569    }
1570
1571    // Some properties are generated as Value, but for which the default constructed Value must be initialized
1572    for (prop_name, decl) in &description.original.root_element.borrow().property_declarations {
1573        if !matches!(
1574            decl.property_type,
1575            Type::Struct { .. } | Type::Array(_) | Type::Enumeration(_)
1576        ) || decl.is_alias.is_some()
1577        {
1578            continue;
1579        }
1580        if let Some(b) = description.original.root_element.borrow().bindings.get(prop_name)
1581            && b.borrow().two_way_bindings.is_empty()
1582        {
1583            continue;
1584        }
1585        let p = description.custom_properties.get(prop_name).unwrap();
1586        unsafe {
1587            let item = Pin::new_unchecked(&*instance_ref.as_ptr().add(p.offset));
1588            p.prop.set(item, eval::default_value_for_type(&decl.property_type), None).unwrap();
1589        }
1590    }
1591
1592    generator::handle_property_bindings_init(
1593        &description.original,
1594        |elem, prop_name, binding| unsafe {
1595            let is_root = Rc::ptr_eq(
1596                elem,
1597                &elem.borrow().enclosing_component.upgrade().unwrap().root_element,
1598            );
1599            let elem = elem.borrow();
1600            let is_const = binding.analysis.as_ref().is_some_and(|a| a.is_const);
1601
1602            let property_type = elem.lookup_property(prop_name).property_type;
1603            if let Type::Function { .. } = property_type {
1604                // function don't need initialization
1605            } else if let Type::Callback { .. } = property_type {
1606                if !matches!(binding.expression, Expression::Invalid) {
1607                    let expr = binding.expression.clone();
1608                    let description = description.clone();
1609                    if let Some(callback_offset) =
1610                        description.custom_callbacks.get(prop_name).filter(|_| is_root)
1611                    {
1612                        let callback = callback_offset.apply(instance_ref.as_ref());
1613                        callback.set_handler(make_callback_eval_closure(expr, self_weak.clone()));
1614                    } else {
1615                        let item_within_component = &description.items[&elem.id];
1616                        let item = item_within_component.item_from_item_tree(instance_ref.as_ptr());
1617                        if let Some(callback) =
1618                            item_within_component.rtti.callbacks.get(prop_name.as_str())
1619                        {
1620                            callback.set_handler(
1621                                item,
1622                                Box::new(make_callback_eval_closure(expr, self_weak.clone())),
1623                            );
1624                        } else {
1625                            panic!("unknown callback {prop_name}")
1626                        }
1627                    }
1628                }
1629            } else if let Some(PropertiesWithinComponent { offset, prop: prop_info, .. }) =
1630                description.custom_properties.get(prop_name).filter(|_| is_root)
1631            {
1632                let is_state_info = matches!(&property_type, Type::Struct (s) if matches!(s.name, StructName::BuiltinPrivate(BuiltinPrivateStruct::StateInfo)));
1633                if is_state_info {
1634                    let prop = Pin::new_unchecked(
1635                        &*(instance_ref.as_ptr().add(*offset)
1636                            as *const Property<i_slint_core::properties::StateInfo>),
1637                    );
1638                    let e = binding.expression.clone();
1639                    let state_binding = make_binding_eval_closure(e, self_weak.clone());
1640                    i_slint_core::properties::set_state_binding(prop, move || {
1641                        state_binding().try_into().unwrap()
1642                    });
1643                    return;
1644                }
1645
1646                let maybe_animation = animation_for_property(instance_ref, &binding.animation);
1647                let item = Pin::new_unchecked(&*instance_ref.as_ptr().add(*offset));
1648
1649                if !matches!(binding.expression, Expression::Invalid) {
1650                    if is_const {
1651                        let v = eval::eval_expression(
1652                            &binding.expression,
1653                            &mut eval::EvalLocalContext::from_component_instance(instance_ref),
1654                        );
1655                        prop_info.set(item, v, None).unwrap();
1656                    } else {
1657                        let e = binding.expression.clone();
1658                        prop_info
1659                            .set_binding(
1660                                item,
1661                                Box::new(make_binding_eval_closure(e, self_weak.clone())),
1662                                maybe_animation,
1663                            )
1664                            .unwrap();
1665                    }
1666                }
1667                for twb in &binding.two_way_bindings {
1668                    if twb.field_access.is_empty()
1669                        && !matches!(&property_type, Type::Struct(..) | Type::Array(..))
1670                    {
1671                        // Safety: The compiler must have ensured that the properties exist and are of the same type
1672                        // (Except for struct and array, which may map to a Value)
1673                        prop_info
1674                            .link_two_ways(item, get_property_ptr(&twb.property, instance_ref));
1675                    } else {
1676                        let (common, map) = prepare_for_two_way_binding(instance_ref, twb);
1677                        prop_info.link_two_way_with_map(item, common, map);
1678                    }
1679                }
1680            } else {
1681                let item_within_component = &description.items[&elem.id];
1682                let item = item_within_component.item_from_item_tree(instance_ref.as_ptr());
1683                if let Some(prop_rtti) =
1684                    item_within_component.rtti.properties.get(prop_name.as_str())
1685                {
1686                    let maybe_animation = animation_for_property(instance_ref, &binding.animation);
1687
1688                    for twb in &binding.two_way_bindings {
1689                        if twb.field_access.is_empty()
1690                            && !matches!(&property_type, Type::Struct(..) | Type::Array(..))
1691                        {
1692                            // Safety: The compiler must have ensured that the properties exist and are of the same type
1693                            prop_rtti
1694                                .link_two_ways(item, get_property_ptr(&twb.property, instance_ref));
1695                        } else {
1696                            let (common, map) = prepare_for_two_way_binding(instance_ref, twb);
1697                            prop_rtti.link_two_way_with_map(item, common, map);
1698                        }
1699                    }
1700                    if !matches!(binding.expression, Expression::Invalid) {
1701                        if is_const {
1702                            prop_rtti
1703                                .set(
1704                                    item,
1705                                    eval::eval_expression(
1706                                        &binding.expression,
1707                                        &mut eval::EvalLocalContext::from_component_instance(
1708                                            instance_ref,
1709                                        ),
1710                                    ),
1711                                    maybe_animation.as_animation(),
1712                                )
1713                                .unwrap();
1714                        } else {
1715                            let e = binding.expression.clone();
1716                            prop_rtti.set_binding(
1717                                item,
1718                                Box::new(make_binding_eval_closure(e, self_weak.clone())),
1719                                maybe_animation,
1720                            );
1721                        }
1722                    }
1723                } else {
1724                    panic!("unknown property {} in {}", prop_name, elem.id);
1725                }
1726            }
1727        },
1728    );
1729
1730    for rep_in_comp in &description.repeater {
1731        generativity::make_guard!(guard);
1732        let rep_in_comp = rep_in_comp.unerase(guard);
1733
1734        let repeater = rep_in_comp.offset.apply_pin(instance_ref.instance);
1735        let expr = rep_in_comp.model.clone();
1736        let model_binding_closure = make_binding_eval_closure(expr, self_weak.clone());
1737        if rep_in_comp.is_conditional {
1738            let bool_model = Rc::new(crate::value_model::BoolModel::default());
1739            repeater.set_model_binding(move || {
1740                let v = model_binding_closure();
1741                bool_model.set_value(v.try_into().expect("condition model is bool"));
1742                ModelRc::from(bool_model.clone())
1743            });
1744        } else {
1745            repeater.set_model_binding(move || {
1746                let m = model_binding_closure();
1747                if let Value::Model(m) = m {
1748                    m
1749                } else {
1750                    ModelRc::new(crate::value_model::ValueModel::new(m))
1751                }
1752            });
1753        }
1754    }
1755    self_rc
1756}

1758fn prepare_for_two_way_binding(
1759    instance_ref: InstanceRef,
1760    twb: &i_slint_compiler::expression_tree::TwoWayBinding,
1761) -> (Pin<Rc<Property<Value>>>, Option<Rc<dyn rtti::TwoWayBindingMapping<Value>>>) {
1762    let element = twb.property.element();
1763    let name = twb.property.name();
1764    generativity::make_guard!(guard);
1765    let enclosing_component = eval::enclosing_component_instance_for_element(
1766        &element,
1767        &eval::ComponentInstance::InstanceRef(instance_ref),
1768        guard,
1769    );
1770    let map: Option<Rc<dyn rtti::TwoWayBindingMapping<Value>>> = if twb.field_access.is_empty() {
1771        None
1772    } else {
1773        struct FieldAccess(Vec<SmolStr>);
1774        impl rtti::TwoWayBindingMapping<Value> for FieldAccess {
1775            fn map_to(&self, value: &Value) -> Value {
1776                let mut value = value.clone();
1777                for f in &self.0 {
1778                    match value {
1779                        Value::Struct(o) => value = o.get_field(f).cloned().unwrap_or_default(),
1780                        Value::Void => return Value::Void,
1781                        _ => panic!("Cannot map to a field of a non-struct {value:?}  - {f}"),
1782                    }
1783                }
1784                value
1785            }
1786            fn map_from(&self, mut value: &mut Value, from: &Value) {

1813                .get(element.id.as_str())

1816                .rtti
1817                .properties

1865pub struct ErasedItemTreeBox(ItemTreeBox<'static>);
1866impl ErasedItemTreeBox {
1867    pub fn unerase<'a, 'id>(
1868        &'a self,
1869        _guard: generativity::Guard<'id>,
1870    ) -> Pin<&'a ItemTreeBox<'id>> {
1871        Pin::new(
1872            //Safety: 'id is unique because of `_guard`
1873            unsafe { core::mem::transmute::<&ItemTreeBox<'static>, &ItemTreeBox<'id>>(&self.0) },
1874        )
1875    }
1876
1877    pub fn borrow(&self) -> ItemTreeRefPin<'_> {
1878        // Safety: it is safe to access self.0 here because the 'id lifetime does not leak
1879        self.0.borrow()
1880    }
1881
1882    pub fn window_adapter_ref(&self) -> Result<&WindowAdapterRc, PlatformError> {
1883        self.0.window_adapter_ref()
1884    }
1885
1886    pub fn run_setup_code(&self) {
1887        generativity::make_guard!(guard);
1888        let compo_box = self.unerase(guard);
1889        let instance_ref = compo_box.borrow_instance();
1890        for extra_init_code in self.0.description.original.init_code.borrow().iter() {
1891            eval::eval_expression(
1892                extra_init_code,
1893                &mut eval::EvalLocalContext::from_component_instance(instance_ref),
1894            );
1895        }
1896        if let Some(cts) = instance_ref.description.change_trackers.as_ref() {
1897            let self_weak = instance_ref.self_weak().get().unwrap();
1898            let v = cts
1899                .1
1900                .iter()
1901                .enumerate()
1902                .map(|(idx, _)| {
1903                    let ct = ChangeTracker::default();
1904                    ct.init(
1905                        self_weak.clone(),
1906                        move |self_weak| {
1907                            let s = self_weak.upgrade().unwrap();
1908                            generativity::make_guard!(guard);
1909                            let compo_box = s.unerase(guard);
1910                            let instance_ref = compo_box.borrow_instance();
1911                            let nr = &s.0.description.change_trackers.as_ref().unwrap().1[idx].0;
1912                            eval::load_property(instance_ref, &nr.element(), nr.name()).unwrap()
1913                        },
1914                        move |self_weak, _| {
1915                            let s = self_weak.upgrade().unwrap();
1916                            generativity::make_guard!(guard);
1917                            let compo_box = s.unerase(guard);
1918                            let instance_ref = compo_box.borrow_instance();
1919                            let e = &s.0.description.change_trackers.as_ref().unwrap().1[idx].1;
1920                            eval::eval_expression(
1921                                e,
1922                                &mut eval::EvalLocalContext::from_component_instance(instance_ref),
1923                            );
1924                        },
1925                    );
1926                    ct
1927                })
1928                .collect::<Vec<_>>();
1929            cts.0
1930                .apply_pin(instance_ref.instance)
1931                .set(v)
1932                .unwrap_or_else(|_| panic!("run_setup_code called twice?"));
1933        }
1934        update_timers(instance_ref);
1935    }
1936}

2407        root_weak: ItemTreeWeak,
2408        do_create: bool,
2409        instance: &'b Instance<'id2>,
2410    ) -> Result<&'b WindowAdapterRc, PlatformError> {
2411        // We are the actual root: Generate and store a window_adapter if necessary
2412        description.window_adapter_offset.apply(instance).get_or_try_init(|| {

2419                // We are embedded: Get window adapter from our parent
2420                let mut result = None;
2421                vtable::VRc::borrow_pin(parent.item_tree())
2422                    .as_ref()
2423                    .window_adapter(do_create, &mut result);
2424                result.ok_or(PlatformError::NoPlatform)
2425            } else if do_create {
2426                let extra_data = description.extra_data_offset.apply(instance);
2427                let window_adapter = // We are the root: Create a window adapter
2428                    i_slint_backend_selector::with_platform(|_b| {

2600pub fn update_timers(instance: InstanceRef) {
2601    let ts = instance.description.original.timers.borrow();
2602    for (desc, offset) in ts.iter().zip(&instance.description.timers) {
2603        let timer = offset.apply(instance.as_ref());
2604        let running =
2605            eval::load_property(instance, &desc.running.element(), desc.running.name()).unwrap();
2606        if matches!(running, Value::Bool(true)) {
2607            let millis: i64 =
2608                eval::load_property(instance, &desc.interval.element(), desc.interval.name())
2609                    .unwrap()
2610                    .try_into()
2611                    .expect("interval must be a duration");
2612            if millis < 0 {
2613                timer.stop();
2614                continue;
2615            }
2616            let interval = core::time::Duration::from_millis(millis as _);
2617            if !timer.running() || interval != timer.interval() {
2618                let callback = desc.triggered.clone();
2619                let self_weak = instance.self_weak().get().unwrap().clone();
2620                timer.start(i_slint_core::timers::TimerMode::Repeated, interval, move || {
2621                    if let Some(instance) = self_weak.upgrade() {
2622                        generativity::make_guard!(guard);
2623                        let c = instance.unerase(guard);
2624                        let c = c.borrow_instance();
2625                        let inst = eval::ComponentInstance::InstanceRef(c);
2626                        eval::invoke_callback(&inst, &callback.element(), callback.name(), &[])
2627                            .unwrap();
2628                    }
2629                });
2630            }
2631        } else {
2632            timer.stop();
2633        }
2634    }
2635}
