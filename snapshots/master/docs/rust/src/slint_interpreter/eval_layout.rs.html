22pub(crate) fn to_runtime(o: Orientation) -> core_layout::Orientation {
23    match o {
24        Orientation::Horizontal => core_layout::Orientation::Horizontal,
25        Orientation::Vertical => core_layout::Orientation::Vertical,
26    }
27}

29pub(crate) fn from_runtime(o: core_layout::Orientation) -> Orientation {
30    match o {
31        core_layout::Orientation::Horizontal => Orientation::Horizontal,
32        core_layout::Orientation::Vertical => Orientation::Vertical,
33    }

36pub(crate) fn compute_grid_layout_info(
37    grid_layout: &GridLayout,
38    organized_data: &GridLayoutOrganizedData,
39    orientation: Orientation,
40    local_context: &mut EvalLocalContext,
41) -> Value {
42    let component = local_context.component_instance;
43    let expr_eval = |nr: &NamedReference| -> f32 {
44        eval::load_property(component, &nr.element(), nr.name()).unwrap().try_into().unwrap()
45    };
46    let (padding, spacing) = padding_and_spacing(&grid_layout.geometry, orientation, &expr_eval);
47    let repeater_indices = grid_repeater_indices(grid_layout, local_context);
48    let constraints = grid_layout_constraints(grid_layout, orientation, local_context);
49    core_layout::grid_layout_info(
50        organized_data.clone(),
51        Slice::from_slice(constraints.as_slice()),
52        Slice::from_slice(repeater_indices.as_slice()),
53        spacing,
54        &padding,
55        to_runtime(orientation),
56    )
57    .into()
58}

60pub(crate) fn compute_layout_info(
61    lay: &Layout,
62    orientation: Orientation,
63    local_context: &mut EvalLocalContext,
64) -> Value {
65    let component = local_context.component_instance;
66    let expr_eval = |nr: &NamedReference| -> f32 {
67        eval::load_property(component, &nr.element(), nr.name()).unwrap().try_into().unwrap()
68    };
69    match lay {
70        Layout::GridLayout(_) => {
71            panic!("only BoxLayout is supported");
72        }
73        Layout::BoxLayout(box_layout) => {
74            let (cells, alignment) =
75                box_layout_data(box_layout, orientation, component, &expr_eval, None);
76            let (padding, spacing) =
77                padding_and_spacing(&box_layout.geometry, orientation, &expr_eval);
78            if orientation == box_layout.orientation {
79                core_layout::box_layout_info(
80                    Slice::from(cells.as_slice()),
81                    spacing,
82                    &padding,
83                    alignment,
84                )
85            } else {
86                core_layout::box_layout_info_ortho(Slice::from(cells.as_slice()), &padding)
87            }
88            .into()
89        }
90    }
91}

93pub(crate) fn organize_grid_layout(
94    layout: &GridLayout,
95    local_context: &mut EvalLocalContext,
96) -> Value {
97    let cells = grid_layout_input_data(layout, local_context);
98    let repeater_indices = grid_repeater_indices(layout, local_context);
99    if let Some(buttons_roles) = &layout.dialog_button_roles {
100        let roles = buttons_roles
101            .iter()
102            .map(|r| DialogButtonRole::from_str(r).unwrap())
103            .collect::<Vec<_>>();
104        core_layout::organize_dialog_button_layout(
105            Slice::from_slice(cells.as_slice()),
106            Slice::from_slice(roles.as_slice()),
107        )
108        .into()
109    } else {
110        core_layout::organize_grid_layout(
111            Slice::from_slice(cells.as_slice()),
112            Slice::from_slice(repeater_indices.as_slice()),
113        )
114        .into()
115    }
116}

118pub(crate) fn solve_grid_layout(
119    organized_data: &GridLayoutOrganizedData,
120    grid_layout: &GridLayout,
121    orientation: Orientation,
122    local_context: &mut EvalLocalContext,
123) -> Value {
124    let component = local_context.component_instance;
125    let expr_eval = |nr: &NamedReference| -> f32 {
126        eval::load_property(component, &nr.element(), nr.name()).unwrap().try_into().unwrap()
127    };
128    let repeater_indices = grid_repeater_indices(grid_layout, local_context);
129    let constraints = grid_layout_constraints(grid_layout, orientation, local_context);
130
131    let (padding, spacing) = padding_and_spacing(&grid_layout.geometry, orientation, &expr_eval);
132    let size_ref = grid_layout.geometry.rect.size_reference(orientation);
133
134    let data = core_layout::GridLayoutData {
135        size: size_ref.map(expr_eval).unwrap_or(0.),
136        spacing,
137        padding,
138        organized_data: organized_data.clone(),
139    };
140
141    core_layout::solve_grid_layout(
142        &data,
143        Slice::from_slice(constraints.as_slice()),
144        to_runtime(orientation),
145        Slice::from_slice(repeater_indices.as_slice()),
146    )
147    .into()
148}

150pub(crate) fn solve_layout(
151    lay: &Layout,
152    orientation: Orientation,
153    local_context: &mut EvalLocalContext,
154) -> Value {
155    let component = local_context.component_instance;
156    let expr_eval = |nr: &NamedReference| -> f32 {
157        eval::load_property(component, &nr.element(), nr.name()).unwrap().try_into().unwrap()
158    };
159
160    match lay {
161        Layout::GridLayout(_) => {
162            panic!("solve_layout called on GridLayout; use solve_grid_layout instead");
163        }
164        Layout::BoxLayout(box_layout) => {
165            let mut repeated_indices = Vec::new();
166            let (cells, alignment) = box_layout_data(
167                box_layout,
168                orientation,
169                component,
170                &expr_eval,
171                Some(&mut repeated_indices),
172            );
173            let (padding, spacing) =
174                padding_and_spacing(&box_layout.geometry, orientation, &expr_eval);
175            let size_ref = match orientation {
176                Orientation::Horizontal => &box_layout.geometry.rect.width_reference,
177                Orientation::Vertical => &box_layout.geometry.rect.height_reference,
178            };
179            core_layout::solve_box_layout(
180                &core_layout::BoxLayoutData {
181                    size: size_ref.as_ref().map(expr_eval).unwrap_or(0.),
182                    spacing,
183                    padding,
184                    alignment,
185                    cells: Slice::from(cells.as_slice()),
186                },
187                Slice::from(repeated_indices.as_slice()),
188            )
189            .into()
190        }
191    }
192}

194fn padding_and_spacing(
195    layout_geometry: &LayoutGeometry,
196    orientation: Orientation,
197    expr_eval: &impl Fn(&NamedReference) -> f32,
198) -> (core_layout::Padding, f32) {
199    let spacing = layout_geometry.spacing.orientation(orientation).map_or(0., expr_eval);
200    let (begin, end) = layout_geometry.padding.begin_end(orientation);
201    let padding =
202        core_layout::Padding { begin: begin.map_or(0., expr_eval), end: end.map_or(0., expr_eval) };
203    (padding, spacing)
204}

206fn repeater_instances(
207    component: InstanceRef,
208    elem: &ElementRc,
209) -> Vec<crate::dynamic_item_tree::DynamicComponentVRc> {
210    generativity::make_guard!(guard);
211    let rep =
212        crate::dynamic_item_tree::get_repeater_by_name(component, elem.borrow().id.as_str(), guard);
213    rep.0.as_ref().ensure_updated(|| {
214        crate::dynamic_item_tree::instantiate(
215            rep.1.clone(),
216            component.self_weak().get().cloned(),
217            None,
218            None,
219            Default::default(),
220        )
221    });
222    rep.0.as_ref().instances_vec()
223}

225fn grid_layout_input_data(
226    grid_layout: &i_slint_compiler::layout::GridLayout,
227    ctx: &EvalLocalContext,
228) -> Vec<core_layout::GridLayoutInputData> {
229    let component = ctx.component_instance;
230    let mut result = Vec::with_capacity(grid_layout.elems.len());
231    let mut after_repeater_in_same_row = false;
232    let mut new_row = true;
233    for elem in grid_layout.elems.iter() {
234        let eval_or_default = |expr: &RowColExpr, component: InstanceRef| match expr {
235            RowColExpr::Literal(value) => *value as f32,
236            RowColExpr::Auto => i_slint_common::ROW_COL_AUTO,
237            RowColExpr::Named(nr) => {
238                // we could check for out-of-bounds here, but organize_grid_layout will also do it
239                eval::load_property(component, &nr.element(), nr.name())
240                    .unwrap()
241                    .try_into()
242                    .unwrap()
243            }
244        };
245
246        let cell_new_row = elem.cell.borrow().new_row;
247        if cell_new_row {
248            after_repeater_in_same_row = false;
249        }
250        if elem.item.element.borrow().repeated.is_some() {
251            let component_vec = repeater_instances(component, &elem.item.element);
252            new_row = cell_new_row;
253            for erased_sub_comp in &component_vec {
254                // Evaluate the row/col/rowspan/colspan expressions in the context of the sub-component
255                generativity::make_guard!(guard);
256                let sub_comp = erased_sub_comp.as_pin_ref();
257                let sub_instance_ref =
258                    unsafe { InstanceRef::from_pin_ref(sub_comp.borrow(), guard) };
259                let row = eval_or_default(&elem.cell.borrow().row_expr, sub_instance_ref);
260                let col = eval_or_default(&elem.cell.borrow().col_expr, sub_instance_ref);
261                let rowspan = eval_or_default(&elem.cell.borrow().rowspan_expr, sub_instance_ref);
262                let colspan = eval_or_default(&elem.cell.borrow().colspan_expr, sub_instance_ref);
263                result.push(core_layout::GridLayoutInputData {
264                    new_row,
265                    col,
266                    row,
267                    colspan,
268                    rowspan,
269                });
270                new_row = false;
271            }
272            after_repeater_in_same_row = true;
273        } else {
274            let new_row =
275                if cell_new_row || !after_repeater_in_same_row { cell_new_row } else { new_row };
276            let row = eval_or_default(&elem.cell.borrow().row_expr, component);
277            let col = eval_or_default(&elem.cell.borrow().col_expr, component);
278            let rowspan = eval_or_default(&elem.cell.borrow().rowspan_expr, component);
279            let colspan = eval_or_default(&elem.cell.borrow().colspan_expr, component);
280            result.push(core_layout::GridLayoutInputData { new_row, col, row, colspan, rowspan });
281        }
282    }
283    result
284}

286fn grid_repeater_indices(
287    grid_layout: &i_slint_compiler::layout::GridLayout,
288    ctx: &mut EvalLocalContext,
289) -> Vec<u32> {
290    let component = ctx.component_instance;
291    let mut repeater_indices = Vec::new();
292
293    let mut num_cells = 0;
294    for cell in grid_layout.elems.iter() {
295        if cell.item.element.borrow().repeated.is_some() {
296            let component_vec = repeater_instances(component, &cell.item.element);
297            repeater_indices.push(num_cells as _);
298            repeater_indices.push(component_vec.len() as _);
299            num_cells += component_vec.len();
300        } else {
301            num_cells += 1;
302        }
303    }
304    repeater_indices
305}

307fn grid_layout_constraints(
308    grid_layout: &i_slint_compiler::layout::GridLayout,
309    orientation: Orientation,
310    ctx: &mut EvalLocalContext,
311) -> Vec<core_layout::LayoutItemInfo> {
312    let component = ctx.component_instance;
313    let expr_eval = |nr: &NamedReference| -> f32 {
314        eval::load_property(component, &nr.element(), nr.name()).unwrap().try_into().unwrap()
315    };
316    let mut constraints = Vec::with_capacity(grid_layout.elems.len());
317
318    for cell in grid_layout.elems.iter() {
319        if cell.item.element.borrow().repeated.is_some() {
320            let component_vec = repeater_instances(component, &cell.item.element);
321            constraints.extend(
322                component_vec
323                    .iter()
324                    .map(|x| x.as_pin_ref().layout_item_info(to_runtime(orientation))),
325            );
326        } else {
327            let mut layout_info = get_layout_info(
328                &cell.item.element,
329                component,
330                &component.window_adapter(),
331                orientation,
332            );
333            fill_layout_info_constraints(
334                &mut layout_info,
335                &cell.item.constraints,
336                orientation,
337                &expr_eval,
338            );
339            constraints.push(core_layout::LayoutItemInfo { constraint: layout_info });
340        }
341    }
342    constraints
343}

345fn box_layout_data(
346    box_layout: &i_slint_compiler::layout::BoxLayout,
347    orientation: Orientation,
348    component: InstanceRef,
349    expr_eval: &impl Fn(&NamedReference) -> f32,
350    mut repeater_indices: Option<&mut Vec<u32>>,
351) -> (Vec<core_layout::LayoutItemInfo>, i_slint_core::items::LayoutAlignment) {
352    let window_adapter = component.window_adapter();
353    let mut cells = Vec::with_capacity(box_layout.elems.len());
354    for cell in &box_layout.elems {
355        if cell.element.borrow().repeated.is_some() {
356            let component_vec = repeater_instances(component, &cell.element);
357            if let Some(ri) = repeater_indices.as_mut() {
358                ri.push(cells.len() as _);
359                ri.push(component_vec.len() as _);
360            }
361            cells.extend(
362                component_vec
363                    .iter()
364                    .map(|x| x.as_pin_ref().layout_item_info(to_runtime(orientation))),
365            );
366        } else {
367            let mut layout_info =
368                get_layout_info(&cell.element, component, &window_adapter, orientation);
369            fill_layout_info_constraints(
370                &mut layout_info,
371                &cell.constraints,
372                orientation,
373                &expr_eval,
374            );
375            cells.push(core_layout::LayoutItemInfo { constraint: layout_info });
376        }
377    }
378    let alignment = box_layout
379        .geometry
380        .alignment
381        .as_ref()
382        .map(|nr| {
383            eval::load_property(component, &nr.element(), nr.name())
384                .unwrap()
385                .try_into()
386                .unwrap_or_default()
387        })
388        .unwrap_or_default();
389    (cells, alignment)
390}

392pub(crate) fn fill_layout_info_constraints(
393    layout_info: &mut core_layout::LayoutInfo,
394    constraints: &LayoutConstraints,
395    orientation: Orientation,
396    expr_eval: &impl Fn(&NamedReference) -> f32,
397) {
398    let is_percent =
399        |nr: &NamedReference| Expression::PropertyReference(nr.clone()).ty() == Type::Percent;
400
401    match orientation {
402        Orientation::Horizontal => {
403            if let Some(e) = constraints.min_width.as_ref() {
404                if !is_percent(e) {
405                    layout_info.min = expr_eval(e)
406                } else {
407                    layout_info.min_percent = expr_eval(e)
408                }
409            }
410            if let Some(e) = constraints.max_width.as_ref() {
411                if !is_percent(e) {
412                    layout_info.max = expr_eval(e)
413                } else {
414                    layout_info.max_percent = expr_eval(e)
415                }
416            }
417            if let Some(e) = constraints.preferred_width.as_ref() {
418                layout_info.preferred = expr_eval(e);
419            }
420            if let Some(e) = constraints.horizontal_stretch.as_ref() {
421                layout_info.stretch = expr_eval(e);
422            }
423        }
424        Orientation::Vertical => {
425            if let Some(e) = constraints.min_height.as_ref() {
426                if !is_percent(e) {
427                    layout_info.min = expr_eval(e)
428                } else {
429                    layout_info.min_percent = expr_eval(e)
430                }
431            }
432            if let Some(e) = constraints.max_height.as_ref() {
433                if !is_percent(e) {
434                    layout_info.max = expr_eval(e)
435                } else {
436                    layout_info.max_percent = expr_eval(e)
437                }
438            }
439            if let Some(e) = constraints.preferred_height.as_ref() {
440                layout_info.preferred = expr_eval(e);
441            }
442            if let Some(e) = constraints.vertical_stretch.as_ref() {
443                layout_info.stretch = expr_eval(e);
444            }
445        }
446    }
447}

449/// Get the layout info for an element based on the layout_info_prop or the builtin item layout_info
450pub(crate) fn get_layout_info(
451    elem: &ElementRc,
452    component: InstanceRef,
453    window_adapter: &Rc<dyn WindowAdapter>,
454    orientation: Orientation,
455) -> core_layout::LayoutInfo {
456    let elem = elem.borrow();
457    if let Some(nr) = elem.layout_info_prop(orientation) {
458        eval::load_property(component, &nr.element(), nr.name()).unwrap().try_into().unwrap()
459    } else {
460        let item = &component
461            .description
462            .items
463            .get(elem.id.as_str())
464            .unwrap_or_else(|| panic!("Internal error: Item {} not found", elem.id));
465        let item_comp = component.self_weak().get().unwrap().upgrade().unwrap();
466
467        unsafe {
468            item.item_from_item_tree(component.as_ptr()).as_ref().layout_info(
469                to_runtime(orientation),
470                window_adapter,
471                &ItemRc::new(vtable::VRc::into_dyn(item_comp), item.item_index()),
472            )
473        }
474    }
