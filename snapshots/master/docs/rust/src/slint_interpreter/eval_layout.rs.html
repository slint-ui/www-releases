22pub(crate) fn to_runtime(o: Orientation) -> core_layout::Orientation {
23    match o {
24        Orientation::Horizontal => core_layout::Orientation::Horizontal,
25        Orientation::Vertical => core_layout::Orientation::Vertical,
26    }
27}

29pub(crate) fn from_runtime(o: core_layout::Orientation) -> Orientation {
30    match o {
31        core_layout::Orientation::Horizontal => Orientation::Horizontal,
32        core_layout::Orientation::Vertical => Orientation::Vertical,
33    }

36pub(crate) fn compute_grid_layout_info(
37    grid_layout: &GridLayout,
38    organized_data: &GridLayoutOrganizedData,
39    orientation: Orientation,
40    local_context: &mut EvalLocalContext,
41) -> Value {
42    let component = local_context.component_instance;
43    let expr_eval = |nr: &NamedReference| -> f32 {
44        eval::load_property(component, &nr.element(), nr.name()).unwrap().try_into().unwrap()
45    };
46    let (padding, spacing) = padding_and_spacing(&grid_layout.geometry, orientation, &expr_eval);
47    let constraints = grid_layout_constraints(grid_layout, orientation, local_context);
48    core_layout::grid_layout_info(
49        organized_data.clone(),
50        Slice::from_slice(constraints.as_slice()),
51        spacing,
52        &padding,
53        to_runtime(orientation),
54    )
55    .into()
56}

58pub(crate) fn compute_layout_info(
59    lay: &Layout,
60    orientation: Orientation,
61    local_context: &mut EvalLocalContext,
62) -> Value {
63    let component = local_context.component_instance;
64    let expr_eval = |nr: &NamedReference| -> f32 {
65        eval::load_property(component, &nr.element(), nr.name()).unwrap().try_into().unwrap()
66    };
67    match lay {
68        Layout::GridLayout(_) => {
69            panic!("only BoxLayout is supported");
70        }
71        Layout::BoxLayout(box_layout) => {
72            let (cells, alignment) =
73                box_layout_data(box_layout, orientation, component, &expr_eval, None);
74            let (padding, spacing) =
75                padding_and_spacing(&box_layout.geometry, orientation, &expr_eval);
76            if orientation == box_layout.orientation {
77                core_layout::box_layout_info(
78                    Slice::from(cells.as_slice()),
79                    spacing,
80                    &padding,
81                    alignment,
82                )
83            } else {
84                core_layout::box_layout_info_ortho(Slice::from(cells.as_slice()), &padding)
85            }
86            .into()
87        }
88    }
89}

91pub(crate) fn organize_grid_layout(
92    layout: &GridLayout,
93    local_context: &mut EvalLocalContext,
94) -> Value {
95    let component = local_context.component_instance;
96    let expr_eval = |nr: &NamedReference| -> f32 {
97        eval::load_property(component, &nr.element(), nr.name()).unwrap().try_into().unwrap()
98    };
99    let cells = grid_layout_input_data(layout, &expr_eval);
100
101    if let Some(buttons_roles) = &layout.dialog_button_roles {
102        let roles = buttons_roles
103            .iter()
104            .map(|r| DialogButtonRole::from_str(r).unwrap())
105            .collect::<Vec<_>>();
106        core_layout::organize_dialog_button_layout(
107            Slice::from_slice(cells.as_slice()),
108            Slice::from_slice(roles.as_slice()),
109        )
110        .into()
111    } else {
112        core_layout::organize_grid_layout(Slice::from_slice(cells.as_slice())).into()
113    }
114}

116pub(crate) fn solve_grid_layout(
117    organized_data: &GridLayoutOrganizedData,
118    grid_layout: &GridLayout,
119    orientation: Orientation,
120    local_context: &mut EvalLocalContext,
121) -> Value {
122    let component = local_context.component_instance;
123    let expr_eval = |nr: &NamedReference| -> f32 {
124        eval::load_property(component, &nr.element(), nr.name()).unwrap().try_into().unwrap()
125    };
126    let constraints = grid_layout_constraints(grid_layout, orientation, local_context);
127
128    let (padding, spacing) = padding_and_spacing(&grid_layout.geometry, orientation, &expr_eval);
129    let size_ref = grid_layout.geometry.rect.size_reference(orientation);
130
131    let data = core_layout::GridLayoutData {
132        size: size_ref.map(expr_eval).unwrap_or(0.),
133        spacing,
134        padding,
135        organized_data: organized_data.clone(),
136    };
137
138    core_layout::solve_grid_layout(
139        &data,
140        Slice::from_slice(constraints.as_slice()),
141        to_runtime(orientation),
142    )
143    .into()
144}

146pub(crate) fn solve_layout(
147    lay: &Layout,
148    orientation: Orientation,
149    local_context: &mut EvalLocalContext,
150) -> Value {
151    let component = local_context.component_instance;
152    let expr_eval = |nr: &NamedReference| -> f32 {
153        eval::load_property(component, &nr.element(), nr.name()).unwrap().try_into().unwrap()
154    };
155
156    match lay {
157        Layout::GridLayout(_) => {
158            panic!("solve_layout called on GridLayout; use solve_grid_layout instead");
159        }
160        Layout::BoxLayout(box_layout) => {
161            let mut repeated_indices = Vec::new();
162            let (cells, alignment) = box_layout_data(
163                box_layout,
164                orientation,
165                component,
166                &expr_eval,
167                Some(&mut repeated_indices),
168            );
169            let (padding, spacing) =
170                padding_and_spacing(&box_layout.geometry, orientation, &expr_eval);
171            let size_ref = match orientation {
172                Orientation::Horizontal => &box_layout.geometry.rect.width_reference,
173                Orientation::Vertical => &box_layout.geometry.rect.height_reference,
174            };
175            core_layout::solve_box_layout(
176                &core_layout::BoxLayoutData {
177                    size: size_ref.as_ref().map(expr_eval).unwrap_or(0.),
178                    spacing,
179                    padding,
180                    alignment,
181                    cells: Slice::from(cells.as_slice()),
182                },
183                Slice::from(repeated_indices.as_slice()),
184            )
185            .into()
186        }
187    }
188}

190fn padding_and_spacing(
191    layout_geometry: &LayoutGeometry,
192    orientation: Orientation,
193    expr_eval: &impl Fn(&NamedReference) -> f32,
194) -> (core_layout::Padding, f32) {
195    let spacing = layout_geometry.spacing.orientation(orientation).map_or(0., expr_eval);
196    let (begin, end) = layout_geometry.padding.begin_end(orientation);
197    let padding =
198        core_layout::Padding { begin: begin.map_or(0., expr_eval), end: end.map_or(0., expr_eval) };
199    (padding, spacing)
200}

202fn repeater_instances(
203    component: InstanceRef,
204    elem: &ElementRc,
205) -> Vec<crate::dynamic_item_tree::DynamicComponentVRc> {
206    generativity::make_guard!(guard);
207    let rep =
208        crate::dynamic_item_tree::get_repeater_by_name(component, elem.borrow().id.as_str(), guard);
209    rep.0.as_ref().ensure_updated(|| {
210        let instance = crate::dynamic_item_tree::instantiate(
211            rep.1.clone(),
212            component.self_weak().get().cloned(),
213            None,
214            None,
215            Default::default(),
216        );
217        instance
218    });
219    rep.0.as_ref().instances_vec()
220}

222fn grid_layout_input_data(
223    grid_layout: &i_slint_compiler::layout::GridLayout,
224    expr_eval: &impl Fn(&NamedReference) -> f32,
225) -> Vec<core_layout::GridLayoutInputData> {
226    grid_layout
227        .elems
228        .iter()
229        .map(|cell| {
230            let eval_or_default = |expr: &RowColExpr| match expr {
231                RowColExpr::Literal(value) => *value,
232                RowColExpr::Named(e) => {
233                    let value = expr_eval(e);
234                    if value >= 0.0 && value <= u16::MAX as f32 {
235                        value as u16
236                    } else {
237                        panic!(
238                            "Expected a positive integer, but got {:?} while evaluating {:?}",
239                            value, e
240                        );
241                    }
242                }
243            };
244            let row = eval_or_default(&cell.row_expr);
245            let col = eval_or_default(&cell.col_expr);
246            let rowspan = eval_or_default(&cell.rowspan_expr);
247            let colspan = eval_or_default(&cell.colspan_expr);
248            core_layout::GridLayoutInputData { new_row: cell.new_row, col, row, colspan, rowspan }
249        })
250        .collect()
251}

253fn grid_layout_constraints(
254    grid_layout: &i_slint_compiler::layout::GridLayout,
255    orientation: Orientation,
256    ctx: &mut EvalLocalContext,
257) -> Vec<core_layout::LayoutInfo> {
258    let component = ctx.component_instance;
259    let expr_eval = |nr: &NamedReference| -> f32 {
260        eval::load_property(component, &nr.element(), nr.name()).unwrap().try_into().unwrap()
261    };
262    let mut constraints = Vec::with_capacity(grid_layout.elems.len());
263
264    for cell in grid_layout.elems.iter() {
265        let mut layout_info = get_layout_info(
266            &cell.item.element,
267            component,
268            &component.window_adapter(),
269            orientation,
270        );
271        fill_layout_info_constraints(
272            &mut layout_info,
273            &cell.item.constraints,
274            orientation,
275            &expr_eval,
276        );
277        constraints.push(layout_info);
278    }
279    constraints
280}

282fn box_layout_data(
283    box_layout: &i_slint_compiler::layout::BoxLayout,
284    orientation: Orientation,
285    component: InstanceRef,
286    expr_eval: &impl Fn(&NamedReference) -> f32,
287    mut repeater_indices: Option<&mut Vec<u32>>,
288) -> (Vec<core_layout::BoxLayoutCellData>, i_slint_core::items::LayoutAlignment) {
289    let window_adapter = component.window_adapter();
290    let mut cells = Vec::with_capacity(box_layout.elems.len());
291    for cell in &box_layout.elems {
292        if cell.element.borrow().repeated.is_some() {
293            let component_vec = repeater_instances(component, &cell.element);
294            if let Some(ri) = repeater_indices.as_mut() {
295                ri.push(cells.len() as _);
296                ri.push(component_vec.len() as _);
297            }
298            cells.extend(
299                component_vec
300                    .iter()
301                    .map(|x| x.as_pin_ref().box_layout_data(to_runtime(orientation))),
302            );
303        } else {
304            let mut layout_info =
305                get_layout_info(&cell.element, component, &window_adapter, orientation);
306            fill_layout_info_constraints(
307                &mut layout_info,
308                &cell.constraints,
309                orientation,
310                &expr_eval,
311            );
312            cells.push(core_layout::BoxLayoutCellData { constraint: layout_info });
313        }
314    }
315    let alignment = box_layout
316        .geometry
317        .alignment
318        .as_ref()
319        .map(|nr| {
320            eval::load_property(component, &nr.element(), nr.name())
321                .unwrap()
322                .try_into()
323                .unwrap_or_default()
324        })
325        .unwrap_or_default();
326    (cells, alignment)
327}

329pub(crate) fn fill_layout_info_constraints(
330    layout_info: &mut core_layout::LayoutInfo,
331    constraints: &LayoutConstraints,
332    orientation: Orientation,
333    expr_eval: &impl Fn(&NamedReference) -> f32,
334) {
335    let is_percent =
336        |nr: &NamedReference| Expression::PropertyReference(nr.clone()).ty() == Type::Percent;
337
338    match orientation {
339        Orientation::Horizontal => {
340            if let Some(e) = constraints.min_width.as_ref() {
341                if !is_percent(e) {
342                    layout_info.min = expr_eval(e)
343                } else {
344                    layout_info.min_percent = expr_eval(e)
345                }
346            }
347            if let Some(e) = constraints.max_width.as_ref() {
348                if !is_percent(e) {
349                    layout_info.max = expr_eval(e)
350                } else {
351                    layout_info.max_percent = expr_eval(e)
352                }
353            }
354            if let Some(e) = constraints.preferred_width.as_ref() {
355                layout_info.preferred = expr_eval(e);
356            }
357            if let Some(e) = constraints.horizontal_stretch.as_ref() {
358                layout_info.stretch = expr_eval(e);
359            }
360        }
361        Orientation::Vertical => {
362            if let Some(e) = constraints.min_height.as_ref() {
363                if !is_percent(e) {
364                    layout_info.min = expr_eval(e)
365                } else {
366                    layout_info.min_percent = expr_eval(e)
367                }
368            }
369            if let Some(e) = constraints.max_height.as_ref() {
370                if !is_percent(e) {
371                    layout_info.max = expr_eval(e)
372                } else {
373                    layout_info.max_percent = expr_eval(e)
374                }
375            }
376            if let Some(e) = constraints.preferred_height.as_ref() {
377                layout_info.preferred = expr_eval(e);
378            }
379            if let Some(e) = constraints.vertical_stretch.as_ref() {
380                layout_info.stretch = expr_eval(e);
381            }
382        }
383    }
384}

386/// Get the layout info for an element based on the layout_info_prop or the builtin item layout_info
387pub(crate) fn get_layout_info(
388    elem: &ElementRc,
389    component: InstanceRef,
390    window_adapter: &Rc<dyn WindowAdapter>,
391    orientation: Orientation,
392) -> core_layout::LayoutInfo {
393    let elem = elem.borrow();
394    if let Some(nr) = elem.layout_info_prop(orientation) {
395        eval::load_property(component, &nr.element(), nr.name()).unwrap().try_into().unwrap()
396    } else {
397        let item = &component
398            .description
399            .items
400            .get(elem.id.as_str())
401            .unwrap_or_else(|| panic!("Internal error: Item {} not found", elem.id));
402        let item_comp = component.self_weak().get().unwrap().upgrade().unwrap();
403
404        unsafe {
405            item.item_from_item_tree(component.as_ptr()).as_ref().layout_info(
406                to_runtime(orientation),
407                window_adapter,
408                &ItemRc::new(vtable::VRc::into_dyn(item_comp), item.item_index()),
409            )
410        }
411    }
