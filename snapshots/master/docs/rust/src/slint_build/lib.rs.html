4/*!
5This crate serves as a companion crate of the slint crate.
6It is meant to allow you to compile the `.slint` files from your `build.rs` script.
7
8The main entry point of this crate is the [`compile()`] function
9
10## Example
11
12In your Cargo.toml:
13
14```toml
15[package]
16...
17build = "build.rs"
18
19[dependencies]
20slint = "1.10.0"
21...
22
23[build-dependencies]
24slint-build = "1.10.0"
25```
26
27In the `build.rs` file:
28
29```ignore
30fn main() {
31    slint_build::compile("ui/hello.slint").unwrap();
32}
33```
34
35Then in your main file
36
37```ignore
38slint::include_modules!();
39fn main() {
40    HelloWorld::new().run();
41}
42```
43*/

60/// The structure for configuring aspects of the compilation of `.slint` markup files to Rust.
61#[derive(Clone)]
62pub struct CompilerConfiguration {
63    config: i_slint_compiler::CompilerConfiguration,
64}

66/// How should the slint compiler embed images and fonts
67///
68/// Parameter of [`CompilerConfiguration::embed_resources()`]
69#[derive(Clone, PartialEq)]
70pub enum EmbedResourcesKind {
71    /// Paths specified in .slint files are made absolute and the absolute
72    /// paths will be used at run-time to load the resources from the file system.
73    AsAbsolutePath,
74    /// The raw files in .slint files are embedded in the application binary.
75    EmbedFiles,
76    /// File names specified in .slint files will be loaded by the Slint compiler,
77    /// optimized for use with the software renderer and embedded in the application binary.
78    EmbedForSoftwareRenderer,
79}
80
81impl Default for CompilerConfiguration {
82    fn default() -> Self {
83        Self {
84            config: i_slint_compiler::CompilerConfiguration::new(
85                i_slint_compiler::generator::OutputFormat::Rust,
86            ),
87        }
88    }
89}

91impl CompilerConfiguration {
92    /// Creates a new default configuration.
93    pub fn new() -> Self {
94        Self::default()
95    }
96
97    /// Create a new configuration that includes sets the include paths used for looking up
98    /// `.slint` imports to the specified vector of paths.
99    #[must_use]
100    pub fn with_include_paths(self, include_paths: Vec<std::path::PathBuf>) -> Self {
101        let mut config = self.config;
102        config.include_paths = include_paths;
103        Self { config }
104    }
105
106    /// Create a new configuration that sets the library paths used for looking up
107    /// `@library` imports to the specified map of paths.
108    ///
109    /// Each library path can either be a path to a `.slint` file or a directory.
110    /// If it's a file, the library is imported by its name prefixed by `@` (e.g.
111    /// `@example`). The specified file is the only entry-point for the library
112    /// and other files from the library won't be accessible from the outside.
113    /// If it's a directory, a specific file in that directory must be specified
114    /// when importing the library (e.g. `@example/widgets.slint`). This allows
115    /// exposing multiple entry-points for a single library.
116    ///
117    /// Compile `ui/main.slint` and specify an "example" library path:
118    /// ```rust,no_run
119    /// let manifest_dir = std::path::PathBuf::from(std::env::var_os("CARGO_MANIFEST_DIR").unwrap());
120    /// let library_paths = std::collections::HashMap::from([(
121    ///     "example".to_string(),
122    ///     manifest_dir.join("third_party/example/ui/lib.slint"),
123    /// )]);
124    /// let config = slint_build::CompilerConfiguration::new().with_library_paths(library_paths);
125    /// slint_build::compile_with_config("ui/main.slint", config).unwrap();
126    /// ```
127    ///
128    /// Import the "example" library in `ui/main.slint`:
129    /// ```slint,ignore
130    /// import { Example } from "@example";
131    /// ```
132    #[must_use]
133    pub fn with_library_paths(self, library_paths: HashMap<String, std::path::PathBuf>) -> Self {
134        let mut config = self.config;
135        config.library_paths = library_paths;
136        Self { config }
137    }
138
139    /// Create a new configuration that selects the style to be used for widgets.
140    #[must_use]
141    pub fn with_style(self, style: String) -> Self {
142        let mut config = self.config;
143        config.style = Some(style);
144        Self { config }
145    }
146
147    /// Selects how the resources such as images and font are processed.
148    ///
149    /// See [`EmbedResourcesKind`]
150    #[must_use]
151    pub fn embed_resources(self, kind: EmbedResourcesKind) -> Self {
152        let mut config = self.config;
153        config.embed_resources = match kind {
154            EmbedResourcesKind::AsAbsolutePath => {
155                i_slint_compiler::EmbedResourcesKind::OnlyBuiltinResources
156            }
157            EmbedResourcesKind::EmbedFiles => {
158                i_slint_compiler::EmbedResourcesKind::EmbedAllResources
159            }
160            EmbedResourcesKind::EmbedForSoftwareRenderer => {
161                i_slint_compiler::EmbedResourcesKind::EmbedTextures
162            }
163        };
164        Self { config }
165    }
166
167    /// Sets the scale factor to be applied to all `px` to `phx` conversions
168    /// as constant value. This is only intended for MCU environments. Use
169    /// in combination with [`Self::embed_resources`] to pre-scale images and glyphs
170    /// accordingly.
171    #[must_use]
172    pub fn with_scale_factor(self, factor: f32) -> Self {
173        let mut config = self.config;
174        config.const_scale_factor = factor as f64;
175        Self { config }
176    }
177
178    /// Configures the compiler to bundle translations when compiling Slint code.
179    ///
180    /// It expects the path to be the root directory of the translation files.
181    ///
182    /// The translation files should be in the gettext `.po` format and follow this pattern:
183    /// `<path>/<lang>/LC_MESSAGES/<crate>.po`
184    #[must_use]
185    pub fn with_bundled_translations(
186        self,
187        path: impl Into<std::path::PathBuf>,
188    ) -> CompilerConfiguration {
189        let mut config = self.config;
190        config.translation_path_bundle = Some(path.into());
191        Self { config }
192    }
193
194    /// Configures the compiler to emit additional debug info when compiling Slint code.
195    ///
196    /// This is the equivalent to setting `SLINT_EMIT_DEBUG_INFO=1` and using the `slint!()` macro
197    /// and is primarily used by `i-slint-backend-testing`.
198    #[doc(hidden)]
199    #[must_use]
200    pub fn with_debug_info(self, enable: bool) -> Self {
201        let mut config = self.config;
202        config.debug_info = enable;
203        Self { config }
204    }
205
206    /// Configures the compiler to use Signed Distance Field (SDF) encoding for fonts.
207    ///
208    /// This flag only takes effect when `embed_resources` is set to [`EmbedResourcesKind::EmbedForSoftwareRenderer`],
209    /// and requires the `sdf-fonts` cargo feature to be enabled.
210    ///
211    /// [SDF](https://en.wikipedia.org/wiki/Signed_distance_function) reduces the binary size by
212    /// using an alternative representation for fonts, trading off some rendering quality
213    /// for a smaller binary footprint.
214    /// Rendering is slower and may result in slightly inferior visual output.
215    /// Use this on systems with limited flash memory.
216    #[cfg(feature = "sdf-fonts")]
217    #[must_use]
218    pub fn with_sdf_fonts(self, enable: bool) -> Self {
219        let mut config = self.config;
220        config.use_sdf_fonts = enable;
221        Self { config }
222    }

225/// Error returned by the `compile` function
226#[derive(derive_more::Error, derive_more::Display, Debug)]
227#[non_exhaustive]
228pub enum CompileError {
229    /// Cannot read environment variable CARGO_MANIFEST_DIR or OUT_DIR. The build script need to be run via cargo.
230    #[display("Cannot read environment variable CARGO_MANIFEST_DIR or OUT_DIR. The build script need to be run via cargo.")]
231    NotRunViaCargo,
232    /// Parse error. The error are printed in the stderr, and also are in the vector
233    #[display("{_0:?}")]
234    CompileError(#[error(not(source))] Vec<String>),
235    /// Cannot write the generated file
236    #[display("Cannot write the generated file: {_0}")]
237    SaveError(std::io::Error),
238}

240struct CodeFormatter<Sink> {
241    indentation: usize,
242    /// We are currently in a string
243    in_string: bool,
244    /// number of bytes after the last `'`, 0 if there was none
245    in_char: usize,
246    /// In string or char, and the previous character was `\\`
247    escaped: bool,
248    sink: Sink,
249}
250
251impl<Sink> CodeFormatter<Sink> {
252    pub fn new(sink: Sink) -> Self {
253        Self { indentation: 0, in_string: false, in_char: 0, escaped: false, sink }
254    }
255}
256
257impl<Sink: Write> Write for CodeFormatter<Sink> {
258    fn write(&mut self, mut s: &[u8]) -> std::io::Result<usize> {
259        let len = s.len();
260        while let Some(idx) = s.iter().position(|c| match c {
261            b'{' if !self.in_string && self.in_char == 0 => {
262                self.indentation += 1;
263                true
264            }
265            b'}' if !self.in_string && self.in_char == 0 => {
266                self.indentation -= 1;
267                true
268            }
269            b';' if !self.in_string && self.in_char == 0 => true,
270            b'"' if !self.in_string && self.in_char == 0 => {
271                self.in_string = true;
272                self.escaped = false;
273                false
274            }
275            b'"' if self.in_string && !self.escaped => {
276                self.in_string = false;
277                false
278            }
279            b'\'' if !self.in_string && self.in_char == 0 => {
280                self.in_char = 1;
281                self.escaped = false;
282                false
283            }
284            b'\'' if !self.in_string && self.in_char > 0 && !self.escaped => {
285                self.in_char = 0;
286                false
287            }
288            b' ' | b'>' if self.in_char > 2 && !self.escaped => {
289                // probably a lifetime
290                self.in_char = 0;
291                false
292            }
293            b'\\' if (self.in_string || self.in_char > 0) && !self.escaped => {
294                self.escaped = true;
295                // no need to increment in_char since \ isn't a single character
296                false
297            }
298            _ if self.in_char > 0 => {
299                self.in_char += 1;
300                self.escaped = false;
301                false
302            }
303            _ => {
304                self.escaped = false;
305                false
306            }
307        }) {
308            let idx = idx + 1;
309            self.sink.write_all(&s[..idx])?;
310            self.sink.write_all(b"\n")?;
311            for _ in 0..self.indentation {
312                self.sink.write_all(b"    ")?;
313            }
314            s = &s[idx..];
315        }
316        self.sink.write_all(s)?;
317        Ok(len)
318    }
319    fn flush(&mut self) -> std::io::Result<()> {
320        self.sink.flush()
321    }
322}

325fn formatter_test() {
326    fn format_code(code: &str) -> String {
327        let mut res = Vec::new();
328        let mut formatter = CodeFormatter::new(&mut res);
329        formatter.write_all(code.as_bytes()).unwrap();
330        String::from_utf8(res).unwrap()
331    }
332
333    assert_eq!(
334        format_code("fn main() { if ';' == '}' { return \";\"; } else { panic!() } }"),
335        r#"fn main() {
336     if ';' == '}' {
337         return ";";
338         }
339     else {
340         panic!() }
341     }
342"#
343    );
344
345    assert_eq!(
346        format_code(r#"fn xx<'lt>(foo: &'lt str) { println!("{}", '\u{f700}'); return Ok(()); }"#),
347        r#"fn xx<'lt>(foo: &'lt str) {
348     println!("{}", '\u{f700}');
349     return Ok(());
350     }
351"#
352    );
353
354    assert_eq!(
355        format_code(r#"fn main() { ""; "'"; "\""; "{}"; "\\"; "\\\""; }"#),

357     "";
358     "'";
359     "\"";
360     "{}";

380/// Compile the `.slint` file and generate rust code for it.
381///
382/// The generated code code will be created in the directory specified by
383/// the `OUT` environment variable as it is expected for build script.
384///
385/// The following line need to be added within your crate in order to include
386/// the generated code.
387/// ```ignore
388/// slint::include_modules!();
389/// ```
390///
391/// The path is relative to the `CARGO_MANIFEST_DIR`.
392///
393/// In case of compilation error, the errors are shown in `stderr`, the error
394/// are also returned in the [`CompileError`] enum. You must `unwrap` the returned
395/// result to make sure that cargo make the compilation fail in case there were
396/// errors when generating the code.
397///
398/// Please check out the documentation of the `slint` crate for more information
399/// about how to use the generated code.
400///
401/// This function can only be called within a build script run by cargo.
402pub fn compile(path: impl AsRef<std::path::Path>) -> Result<(), CompileError> {
403    compile_with_config(path, CompilerConfiguration::default())
404}
405
406/// Same as [`compile`], but allow to specify a configuration.
407///
408/// Compile `ui/hello.slint` and select the "material" style:
409/// ```rust,no_run
410/// let config =
411///     slint_build::CompilerConfiguration::new()
412///     .with_style("material".into());
413/// slint_build::compile_with_config("ui/hello.slint", config).unwrap();
414/// ```
415pub fn compile_with_config(
416    relative_slint_file_path: impl AsRef<std::path::Path>,
417    config: CompilerConfiguration,
418) -> Result<(), CompileError> {
419    let path = Path::new(&env::var_os("CARGO_MANIFEST_DIR").ok_or(CompileError::NotRunViaCargo)?)
420        .join(relative_slint_file_path.as_ref());
421
422    let absolute_rust_output_file_path =
423        Path::new(&env::var_os("OUT_DIR").ok_or(CompileError::NotRunViaCargo)?).join(
424            path.file_stem()
425                .map(Path::new)
426                .unwrap_or_else(|| Path::new("slint_out"))
427                .with_extension("rs"),
428        );
429
430    let paths_dependencies =
431        compile_with_output_path(path, absolute_rust_output_file_path.clone(), config)?;
432
433    for path_dependency in paths_dependencies {
434        println!("cargo:rerun-if-changed={}", path_dependency.display());
435    }
436
437    println!("cargo:rerun-if-env-changed=SLINT_STYLE");
438    println!("cargo:rerun-if-env-changed=SLINT_FONT_SIZES");
439    println!("cargo:rerun-if-env-changed=SLINT_SCALE_FACTOR");
440    println!("cargo:rerun-if-env-changed=SLINT_ASSET_SECTION");
441    println!("cargo:rerun-if-env-changed=SLINT_EMBED_RESOURCES");
442    println!("cargo:rerun-if-env-changed=SLINT_EMIT_DEBUG_INFO");
443
444    println!(
445        "cargo:rustc-env=SLINT_INCLUDE_GENERATED={}",
446        absolute_rust_output_file_path.display()
447    );
448
449    Ok(())
450}

452/// Similar to [`compile_with_config`], but meant to be used independently of cargo.
453///
454/// Will compile the input file and write the result in the given output file.
455///
456/// Both input_slint_file_path and output_rust_file_path should be absolute paths.
457///
458/// Doesn't print any cargo messages.
459///
460/// Returns a list of all input files that were used to generate the output file. (dependencies)
461pub fn compile_with_output_path(
462    input_slint_file_path: impl AsRef<std::path::Path>,
463    output_rust_file_path: impl AsRef<std::path::Path>,
464    config: CompilerConfiguration,
465) -> Result<Vec<std::path::PathBuf>, CompileError> {
466    let mut diag = BuildDiagnostics::default();
467    let syntax_node = i_slint_compiler::parser::parse_file(&input_slint_file_path, &mut diag);
468
469    if diag.has_errors() {
470        let vec = diag.to_string_vec();
471        diag.print();
472        return Err(CompileError::CompileError(vec));
473    }
474
475    let mut compiler_config = config.config;
476    compiler_config.translation_domain = std::env::var("CARGO_PKG_NAME").ok();
477
478    let syntax_node = syntax_node.expect("diags contained no compilation errors");
479
480    // 'spin_on' is ok here because the compiler in single threaded and does not block if there is no blocking future
481    let (doc, diag, loader) =
482        spin_on::spin_on(i_slint_compiler::compile_syntax_node(syntax_node, diag, compiler_config));
483
484    if diag.has_errors() {
485        let vec = diag.to_string_vec();
486        diag.print();
487        return Err(CompileError::CompileError(vec));
488    }
489
490    let output_file =
491        std::fs::File::create(&output_rust_file_path).map_err(CompileError::SaveError)?;
492    let mut code_formatter = CodeFormatter::new(BufWriter::new(output_file));
493    let generated = i_slint_compiler::generator::rust::generate(&doc, &loader.compiler_config)
494        .map_err(|e| CompileError::CompileError(vec![e.to_string()]))?;
495
496    let mut dependencies: Vec<std::path::PathBuf> = Vec::new();
497
498    for x in &diag.all_loaded_files {
499        if x.is_absolute() {
500            dependencies.push(x.clone());
501        }
502    }
503
504    // print warnings
505    diag.diagnostics_as_string().lines().for_each(|w| {
506        if !w.is_empty() {
507            println!("cargo:warning={}", w.strip_prefix("warning: ").unwrap_or(w))
508        }
509    });
510
511    write!(code_formatter, "{generated}").map_err(CompileError::SaveError)?;
512    dependencies.push(input_slint_file_path.as_ref().to_path_buf());
513
514    for resource in doc.embedded_file_resources.borrow().keys() {
515        if !resource.starts_with("builtin:") {
516            dependencies.push(Path::new(resource).to_path_buf());
517        }
518    }
519
520    Ok(dependencies)
521}

523/// This function is for use the application's build script, in order to print any device specific
524/// build flags reported by the backend
525pub fn print_rustc_flags() -> std::io::Result<()> {
526    if let Some(board_config_path) =
527        std::env::var_os("DEP_MCU_BOARD_SUPPORT_BOARD_CONFIG_PATH").map(std::path::PathBuf::from)
528    {
529        let config = std::fs::read_to_string(board_config_path.as_path())?;
530        let toml = config.parse::<toml_edit::DocumentMut>().expect("invalid board config toml");
531
532        for link_arg in
533            toml.get("link_args").and_then(toml_edit::Item::as_array).into_iter().flatten()
534        {
535            if let Some(option) = link_arg.as_str() {
536                println!("cargo:rustc-link-arg={option}");
537            }
538        }
539
540        for link_search_path in
541            toml.get("link_search_path").and_then(toml_edit::Item::as_array).into_iter().flatten()
542        {
543            if let Some(mut path) = link_search_path.as_str().map(std::path::PathBuf::from) {
544                if path.is_relative() {
545                    path = board_config_path.parent().unwrap().join(path);
546                }
547                println!("cargo:rustc-link-search={}", path.to_string_lossy());
548            }
549        }
550        println!("cargo:rerun-if-env-changed=DEP_MCU_BOARD_SUPPORT_MCU_BOARD_CONFIG_PATH");
551        println!("cargo:rerun-if-changed={}", board_config_path.display());
552    }
553
554    Ok(())
